18macro_rules! expr_prec_layer {
19    ( $name:ident, $inner:ident, $op:expr ) => {
20        fn $name(i: &mut InputStream<'a, 'l>) -> ParseResult<'a, WithSpan<Box<Self>>> {
21            expr_prec_layer(i, Expr::$inner, |i: &mut _| $op.parse_next(i))

28fn expr_prec_layer<'a: 'l, 'l>(
29    i: &mut InputStream<'a, 'l>,
30    inner: fn(&mut InputStream<'a, 'l>) -> ParseResult<'a, WithSpan<Box<Expr<'a>>>>,
31    op: fn(&mut InputStream<'a, 'l>) -> ParseResult<'a>,
32) -> ParseResult<'a, WithSpan<Box<Expr<'a>>>> {
33    let mut expr = inner(i)?;
34
35    let mut level_guard = i.state.level.guard();
36    let mut next = opt(|i: &mut _| {
37        // We need to make sure that we decrement the level before we enter the right-hand side.
38        let i_before = *i;
39        let op = ws(op.with_span()).parse_next(i)?;
40        level_guard.nest(&i_before)?;
41        Ok((op, inner(i)?))
42    });
43    while let Some(((op, span), rhs)) = next.parse_next(i)? {
44        expr = WithSpan::new(Box::new(Expr::BinOp(BinOp { op, lhs: expr, rhs })), span);
45    }
46
47    Ok(expr)
48}

56fn check_expr<'a>(expr: &WithSpan<Box<Expr<'a>>>, allowed: Allowed) -> ParseResult<'a, ()> {
57    match &*expr.inner {
58        &Expr::Var(name) => {
59            // List can be found in rust compiler "can_be_raw" function (although in our case, it's
60            // also used in cases like `match`, so `self` is allowed in this case).
61            if (!allowed.super_keyword && name == "super") || matches!(name, "crate" | "Self") {
62                err_reserved_identifier(&WithSpan::new(name, expr.span))
63            } else if !allowed.underscore && name == "_" {
64                err_underscore_identifier(&WithSpan::new(name, expr.span))
65            } else {
66                Ok(())
67            }
68        }
69        &Expr::IsDefined(var) | &Expr::IsNotDefined(var) => {
70            if var == "_" {
71                err_underscore_identifier(&WithSpan::new(var, expr.span))
72            } else {
73                Ok(())
74            }
75        }
76        Expr::Path(path) => {
77            if let [arg] = path.as_slice()
78                && !crate::can_be_variable_name(*arg.name)
79            {
80                return err_reserved_identifier(&arg.name);
81            }
82            Ok(())
83        }
84        Expr::Array(elems) | Expr::Tuple(elems) | Expr::Concat(elems) => {
85            for elem in elems {
86                check_expr(elem, allowed)?;
87            }
88            Ok(())
89        }
90        Expr::ArrayRepeat(elem, count) => {
91            check_expr(elem, allowed)?;
92            check_expr(count, allowed)?;
93            Ok(())
94        }
95        Expr::AssociatedItem(elem, associated_item) => {
96            if *associated_item.name == "_" {
97                err_underscore_identifier(&associated_item.name)
98            } else if !crate::can_be_variable_name(*associated_item.name) {
99                err_reserved_identifier(&associated_item.name)
100            } else {
101                check_expr(elem, Allowed::default())
102            }
103        }
104        Expr::Index(elem1, elem2) => {
105            check_expr(elem1, Allowed::default())?;
106            check_expr(elem2, Allowed::default())
107        }
108        Expr::BinOp(v) => {
109            check_expr(&v.lhs, Allowed::default())?;
110            check_expr(&v.rhs, Allowed::default())
111        }
112        Expr::Range(v) => {
113            if let Some(elem1) = v.lhs.as_ref() {
114                check_expr(elem1, Allowed::default())?;
115            }
116            if let Some(elem2) = v.rhs.as_ref() {
117                check_expr(elem2, Allowed::default())?;
118            }
119            Ok(())
120        }
121        Expr::As(elem, _)
122        | Expr::Unary(_, elem)
123        | Expr::Group(elem)
124        | Expr::NamedArgument(_, elem)
125        | Expr::Try(elem) => check_expr(elem, Allowed::default()),
126        Expr::Call(v) => {
127            check_expr(
128                &v.path,
129                Allowed {
130                    underscore: false,
131                    super_keyword: true,
132                },
133            )?;
134            for arg in &v.args {
135                check_expr(arg, Allowed::default())?;
136            }
137            Ok(())
138        }
139        Expr::Filter(filter) => {
140            for arg in &filter.arguments {
141                check_expr(arg, Allowed::default())?;
142            }
143            Ok(())
144        }
145        Expr::Struct(s) => {
146            check_expr(
147                &s.path,
148                Allowed {
149                    underscore: false,
150                    super_keyword: true,
151                },
152            )?;
153            for field in &s.fields {
154                if field.name.inner == "_" {
155                    return err_underscore_identifier(&field.name);
156                } else if !crate::can_be_variable_name(field.name.inner) {
157                    return err_reserved_identifier(&field.name);
158                }
159                if let Some(ref value) = field.value {
160                    check_expr(value, Allowed::default())?;
161                }
162            }
163            Ok(())
164        }
165        Expr::LetCond(cond) => check_expr(&cond.expr, Allowed::default()),
166        Expr::ArgumentPlaceholder => cut_error!("unreachable", expr.span),
167        Expr::BoolLit(_)
168        | Expr::NumLit(_, _)
169        | Expr::StrLit(_)
170        | Expr::CharLit(_)
171        | Expr::RustMacro(_, _)
172        | Expr::FilterSource => Ok(()),
173    }
174}

283    pub(super) fn arguments(
284        i: &mut InputStream<'a, 'l>,
285    ) -> ParseResult<'a, WithSpan<Vec<WithSpan<Box<Self>>>>> {
286        fn comma<'a: 'l, 'l>(i: &mut InputStream<'a, 'l>) -> ParseResult<'a, ()> {
287            (ws(','), no_comma).void().parse_next(i)
288        }
289
290        fn no_comma<'a: 'l, 'l>(i: &mut InputStream<'a, 'l>) -> ParseResult<'a, ()> {
291            if let Some(span) = opt(','.span()).parse_next(i)? {
292                cut_error!(
293                    "expected an expression, found a comma in argument list",
294                    span
295                )
296            } else {
297                Ok(())
298            }
299        }
300
301        let span = terminated(ws('('.span()), no_comma).parse_next(i)?;
302
303        // The stack footprint of this function is huge. Effectively, we half the maximum nesting
304        // level of function calls `a(b(c(d(..))))` to make sure not to exceed the stack limit.
305        let mut _level_guard = i.state.level.nest_multiple(i, 2)?;
306
307        let mut named_arguments = HashSet::default();
308        let arguments = separated(
309            1..,
310            move |i: &mut _| {
311                // Needed to prevent borrowing it twice between this closure and the one
312                // calling `Self::named_arguments`.
313                let named_arguments = &mut named_arguments;
314                let has_named_arguments = !named_arguments.is_empty();
315
316                let mut p = alt((
317                    move |i: &mut _| Self::named_argument(i, named_arguments),
318                    move |i: &mut _| Self::parse(i, false),
319                ));
320                let expr = p.parse_next(i)?;
321                if has_named_arguments && !matches!(**expr, Self::NamedArgument(..)) {
322                    return cut_error!("named arguments must always be passed last", expr.span);
323                }
324                Ok(expr)
325            },
326            comma,
327        );
328
329        let (args, closed) =
330            cut_err((opt(terminated(arguments, opt(comma))), opt(ws(')')))).parse_next(i)?;
331        if closed.is_none() {
332            cut_error!("matching closing `)` is missing", span)
333        } else {
334            Ok(WithSpan::new(args.unwrap_or_default(), span))
335        }
336    }

338    fn named_argument(
339        i: &mut InputStream<'a, 'l>,
340        named_arguments: &mut HashSet<&'a str>,
341    ) -> ParseResult<'a, WithSpan<Box<Self>>> {
342        let (((argument, arg_span), _, value), span) =
343            (identifier.with_span(), ws('='), move |i: &mut _| {
344                Self::parse(i, false)
345            })
346                .with_span()
347                .parse_next(i)?;
348        if !named_arguments.insert(argument) {
349            return cut_error!(
350                format!(
351                    "named argument `{}` was passed more than once",
352                    argument.escape_debug()
353                ),
354                arg_span,
355            );
356        }
357
358        Ok(WithSpan::new(
359            Box::new(Self::NamedArgument(
360                WithSpan::new(argument, arg_span),
361                value,
362            )),
363            span,
364        ))
365    }

379    fn range(
380        i: &mut InputStream<'a, 'l>,
381        allow_underscore: bool,
382    ) -> ParseResult<'a, WithSpan<Box<Self>>> {
383        let range_right = move |i: &mut InputStream<'a, 'l>| {
384            let ((op, span), rhs) =
385                (ws(alt(("..=", "..")).with_span()), opt(Self::or)).parse_next(i)?;
386            Ok((op, rhs, span))
387        };
388
389        // `..expr` or `..`
390        let range_to = range_right.map(move |(op, rhs, span)| {
391            WithSpan::new(Box::new(Self::Range(Range { op, lhs: None, rhs })), span)
392        });
393
394        // `expr..expr` or `expr..`
395        let range_from = (Self::or, opt(range_right)).map(move |(lhs, rhs)| match rhs {
396            Some((op, rhs, span)) => WithSpan::new(
397                Box::new(Self::Range(Range {
398                    op,
399                    lhs: Some(lhs),
400                    rhs,
401                })),
402                span,
403            ),
404            None => lhs,
405        });
406
407        let expr = alt((range_to, range_from)).parse_next(i)?;
408        check_expr(
409            &expr,
410            Allowed {
411                underscore: allow_underscore,
412                super_keyword: false,
413            },
414        )?;
415        Ok(expr)
416    }

421    fn compare(i: &mut InputStream<'a, 'l>) -> ParseResult<'a, WithSpan<Box<Self>>> {
422        let mut parse_op = ws(alt(("==", "!=", ">=", ">", "<=", "<")).with_span());
423
424        let (expr, rhs) = (Self::bor, opt((parse_op.by_ref(), Self::bor))).parse_next(i)?;
425        let Some(((op, span), rhs)) = rhs else {
426            return Ok(expr);
427        };
428        let expr = WithSpan::new(Box::new(Expr::BinOp(BinOp { op, lhs: expr, rhs })), span);
429
430        if let Some((op2, span)) = opt(parse_op).parse_next(i)? {
431            return cut_error!(
432                format!(
433                    "comparison operators cannot be chained; \
434                    consider using explicit parentheses, e.g.  `(_ {op} _) {op2} _`"
435                ),
436                span,
437            );
438        }
439
440        Ok(expr)
441    }

449    fn concat(i: &mut InputStream<'a, 'l>) -> ParseResult<'a, WithSpan<Box<Self>>> {
450        #[allow(clippy::type_complexity)]
451        fn concat_expr<'a: 'l, 'l>(
452            i: &mut InputStream<'a, 'l>,
453        ) -> ParseResult<'a, Option<(WithSpan<Box<Expr<'a>>>, std::ops::Range<usize>)>> {
454            let ws1 = |i: &mut _| opt(skip_ws1).parse_next(i);
455            let tilde = (ws1, '~', ws1).with_span();
456            let data = opt((tilde, Expr::muldivmod)).parse_next(i)?;
457
458            let Some((((t1, _, t2), span), expr)) = data else {
459                return Ok(None);
460            };
461            if t1.is_none() || t2.is_none() {
462                return cut_error!("the concat operator `~` must be surrounded by spaces", span);
463            }
464
465            Ok(Some((expr, span)))
466        }
467
468        let expr = Self::muldivmod(i)?;
469        let expr2 = concat_expr(i)?;
470        if let Some((expr2, span)) = expr2 {
471            let mut exprs = vec![expr, expr2];
472            while let Some((expr, _)) = concat_expr(i)? {
473                exprs.push(expr);
474            }
475            Ok(WithSpan::new(Box::new(Self::Concat(exprs)), span))
476        } else {
477            Ok(expr)
478        }
479    }

483    fn is_as(i: &mut InputStream<'a, 'l>) -> ParseResult<'a, WithSpan<Box<Self>>> {
484        let lhs = Self::filtered(i)?;
485        let checkpoint = i.checkpoint();
486        let rhs = opt(ws(identifier.with_span())).parse_next(i)?;
487        match rhs {
488            Some(("is", span)) => Self::is_as_handle_is(i, lhs, span),
489            Some(("as", span)) => Self::is_as_handle_as(i, lhs, span),
490            _ => {
491                i.reset(&checkpoint);
492                Ok(lhs)
493            }
494        }
495    }

497    fn is_as_handle_is(
498        i: &mut InputStream<'a, 'l>,
499        lhs: WithSpan<Box<Expr<'a>>>,
500        span: std::ops::Range<usize>,
501    ) -> ParseResult<'a, WithSpan<Box<Self>>> {
502        let mut rhs = opt(terminated(opt(keyword("not")), ws(keyword("defined"))));
503        let ctor = match rhs.parse_next(i)? {
504            None => {
505                return cut_error!("expected `defined` or `not defined` after `is`", span);
506            }
507            Some(None) => Self::IsDefined,
508            Some(Some(_)) => Self::IsNotDefined,
509        };
510        let var_name = match &**lhs {
511            Self::Var(var_name) => var_name,
512            Self::AssociatedItem(_, _) => {
513                return cut_error!(
514                    "`is defined` operator can only be used on variables, not on their fields",
515                    span,
516                );
517            }
518            _ => {
519                return cut_error!("`is defined` operator can only be used on variables", span);
520            }
521        };
522        Ok(WithSpan::new(Box::new(ctor(var_name)), span))
523    }

525    fn is_as_handle_as(
526        i: &mut InputStream<'a, 'l>,
527        lhs: WithSpan<Box<Expr<'a>>>,
528        span: std::ops::Range<usize>,
529    ) -> ParseResult<'a, WithSpan<Box<Self>>> {
530        let target = opt(path_or_identifier).parse_next(i)?;
531        let Some(PathOrIdentifier::Identifier(target)) = target else {
532            return cut_error!(
533                "`as` operator expects the name of a primitive type on its right-hand side, \
534                not a path or alias",
535                span,
536            );
537        };
538
539        if crate::PRIMITIVE_TYPES.contains(&target) {
540            Ok(WithSpan::new(Box::new(Self::As(lhs, target)), span))
541        } else if target.is_empty() {
542            cut_error!(
543                "`as` operator expects the name of a primitive type on its right-hand side",
544                span,
545            )
546        } else {
547            cut_error!(
548                format!(
549                    "`as` operator expects the name of a primitive type on its right-hand \
550                    side, found `{}`",
551                    target.escape_debug()
552                ),
553                span,
554            )
555        }
556    }

558    fn filtered(i: &mut InputStream<'a, 'l>) -> ParseResult<'a, WithSpan<Box<Self>>> {
559        let mut res = Self::prefix(i)?;
560
561        let mut level_guard = i.state.level.guard();
562        let mut i_before = *i;
563        while let Some((mut filter, span)) = opt(ws(filter.with_span())).parse_next(i)? {
564            level_guard.nest(&i_before)?;
565            filter.arguments.insert(0, res);
566            res = WithSpan::new(Box::new(Self::Filter(filter)), span);
567            i_before = *i;
568        }
569        Ok(res)
570    }

572    fn prefix(i: &mut InputStream<'a, 'l>) -> ParseResult<'a, WithSpan<Box<Self>>> {
573        // This is a rare place where we create recursion in the parsed AST
574        // without recursing the parser call stack. However, this can lead
575        // to stack overflows in drop glue when the AST is very deep.
576        let mut level_guard = i.state.level.guard();
577        let mut i_before = *i;
578        let mut ops = vec![];
579        while let Some(op) = opt(ws(alt(("!", "-", "*", "&")).with_span())).parse_next(i)? {
580            level_guard.nest(&i_before)?;
581            ops.push(op);
582            i_before = *i;
583        }
584
585        let mut expr = Suffix::parse(i)?;
586        for (op, span) in ops.into_iter().rev() {
587            expr = WithSpan::new(Box::new(Self::Unary(op, expr)), span);
588        }
589
590        Ok(expr)
591    }

620    fn group_actually_inner(i: &mut InputStream<'a, 'l>) -> ParseResult<'a, Box<Self>> {
621        let (expr, comma, closing) = (
622            ws(opt(|i: &mut _| Self::parse(i, true))),
623            opt(terminated(','.span(), skip_ws0)),
624            opt(')'),
625        )
626            .parse_next(i)?;
627
628        let expr = match (expr, comma, closing) {
629            // `(expr,`
630            (Some(expr), Some(_), None) => expr,
631            // `()`
632            (None, None, Some(_)) => return Ok(Box::new(Self::Tuple(vec![]))),
633            // `(expr)`
634            (Some(expr), None, Some(_)) => return Ok(Box::new(Self::Group(expr))),
635            // `(expr,)`
636            (Some(expr), Some(_), Some(_)) => return Ok(Box::new(Self::Tuple(vec![expr]))),
637            // `(`
638            (None, None, None) => return cut_error!("expected closing `)` or an expression", *i),
639            // `(expr`
640            (Some(_), None, None) => return cut_error!("expected `,` or `)`", *i),
641            // `(,`
642            (None, Some(span), _) => return cut_error!("stray comma after opening `(`", span),
643        };
644
645        let mut exprs = vec![expr];
646        let collect_items = opt(separated(
647            1..,
648            |i: &mut _| {
649                exprs.push(Self::parse(i, true)?);
650                Ok(())
651            },
652            ws(','),
653        )
654        .map(|()| ()));
655
656        let ((items, comma, close), span) = cut_err((collect_items, ws(opt(',')), opt(')')))
657            .with_span()
658            .parse_next(i)?;
659        let msg = if items.is_none() {
660            "expected `)` or an expression"
661        } else if close.is_some() {
662            return Ok(Box::new(Self::Tuple(exprs)));
663        } else if comma.is_some() {
664            "expected `)` or an expression"
665        } else {
666            "expected `,` or `)`"
667        };
668        cut_error!(msg, span)
669    }

671    fn array(i: &mut InputStream<'a, 'l>) -> ParseResult<'a, WithSpan<Box<Self>>> {
672        let _level_guard = i.state.level.nest(i)?;
673        let (span, mut elements): (_, Vec<_>) = (
674            '['.span(),
675            separated(0.., ws(move |i: &mut _| Self::parse(i, true)), ','),
676        )
677            .parse_next(i)?;
678
679        let expr = if let Some(semicolon) = opt(ws(';'.span())).parse_next(i)? {
680            // array repeat [<el_expr>; <cnt_expr>]
681            let Some(elem) = elements.pop() else {
682                return cut_error!(
683                    "expected element expression for array repeat syntax",
684                    semicolon
685                );
686            };
687            if !elements.is_empty() {
688                return cut_error!("unexpected `;` after expression", semicolon);
689            }
690            let Some(count) = opt(ws(move |i: &mut _| Expr::parse(i, true))).parse_next(i)? else {
691                return cut_error!(
692                    "expected count expression for array repeat syntax after `;`",
693                    semicolon
694                );
695            };
696            if let Some((delim, span)) = ws(opt(one_of((',', ';')).with_span())).parse_next(i)? {
697                return cut_error!(
698                    format!(
699                        "unexpected delimiter `{}`.\n\
700                        Use nested syntax to write a multi-dimensional array: [[expr; N]; M]",
701                        delim.escape_debug()
702                    ),
703                    span
704                );
705            }
706            Self::ArrayRepeat(elem, count)
707        } else {
708            // normal array [<expr>,...?]
709            if !elements.is_empty() {
710                // strip optional trailing comma
711                ws(opt(',')).parse_next(i)?;
712            }
713            Self::Array(elements)
714        };
715
716        if ws(opt(']')).parse_next(i)?.is_none() {
717            let (next, span) = match opt(any_rust_token.with_span()).parse_next(i)? {
718                Some((next, span)) => (Some(next), span),
719                None => (None, span),
720            };
721            return cut_error!(
722                match next {
723                    Some(delim @ (")" | "}")) => format!(
724                        "mismatched closing delimiter `{}`, expected `]`",
725                        delim.escape_debug()
726                    ),
727                    Some(token) =>
728                        format!("unexpected token `{}`, expected `]`", token.escape_debug()),
729                    _ => "missing closing delimiter `]`".to_owned(),
730                },
731                span
732            );
733        }
734        Ok(WithSpan::new(Box::new(expr), span))
735    }

870    fn parse(i: &mut InputStream<'a, 'l>) -> ParseResult<'a, WithSpan<Box<Expr<'a>>>> {
871        let mut level_guard = i.state.level.guard();
872        let mut expr = Expr::single(i)?;
873        let mut right = opt(alt((
874            Self::associated_item,
875            Self::index,
876            Self::call,
877            Self::r#try,
878            Self::r#macro,
879            Self::r#struct,
880        )));
881
882        let mut i_before = i.checkpoint();
883        while let Some(suffix) = right.parse_next(i)? {
884            level_guard.nest(i)?;
885            let (suffix, span) = suffix.deconstruct();
886            let inner = match suffix {
887                Self::AssociatedItem(associated_item) => {
888                    Box::new(Expr::AssociatedItem(expr, associated_item))
889                }
890                Self::Index(index) => Box::new(Expr::Index(expr, index)),
891                Self::Call { generics, args } => Box::new(Expr::Call(Call {
892                    path: expr,
893                    generics,
894                    args,
895                })),
896                Self::Struct { fields, base } => Box::new(Expr::Struct(ExprStruct {
897                    path: expr,
898                    fields,
899                    base,
900                })),
901                Self::Try => Box::new(Expr::Try(expr)),
902                Self::MacroCall(args) => {
903                    let args = WithSpan::new(args, span);
904                    match *expr.inner {
905                        Expr::Path(path) => {
906                            let last = path.last().unwrap();
907                            ensure_macro_name(&last.name)?;
908
909                            if let Some(r) = path.iter().find_map(|r| r.generics.as_ref()) {
910                                return Err(ErrorContext::new(
911                                    "macro paths cannot have generics",
912                                    r.span,
913                                )
914                                .cut());
915                            }
916
917                            Box::new(Expr::RustMacro(
918                                path.into_iter()
919                                    .map(|c: PathComponent<'_>| c.name)
920                                    .collect(),
921                                args,
922                            ))
923                        }
924                        Expr::Var(name) => {
925                            let name = WithSpan::new(name, expr.span);
926                            ensure_macro_name(&name)?;
927                            Box::new(Expr::RustMacro(vec![name], args))
928                        }
929                        _ => {
930                            i.reset(&i_before);
931                            return fail(i);
932                        }
933                    }
934                }
935            };
936            expr = WithSpan::new(inner, span);
937            i_before = i.checkpoint();
938        }
939        Ok(expr)
940    }

968            i: &mut InputStream<'a, 'l>,
969            open_token: Group,
970        ) -> ParseResult<'a, Suffix<'a>> {
971            fn inner<'a: 'l, 'l>(
972                i: &mut InputStream<'a, 'l>,
973                open_token: Group,
974            ) -> ParseResult<'a, <InputStream<'a, 'l> as Stream>::Checkpoint> {
975                let mut open_list = vec![open_token];
976                loop {
977                    let before = i.checkpoint();
978                    let token = ws(opt(token.with_span())).parse_next(i)?;
979                    let after = i.checkpoint();
980                    let Some((token, span)) = token else {
981                        return cut_error!("expected valid tokens in macro call", *i);
982                    };
983                    let close_token = match token {
984                        Token::SomeOther => continue,
985                        Token::Open(group) => {
986                            open_list.push(group);
987                            continue;
988                        }
989                        Token::Close(close_token) => close_token,
990                    };
991                    let open_token = open_list.pop().unwrap();
992
993                    if open_token != close_token {
994                        return cut_error!(
995                            format!(
996                                "expected `{}` but found `{}`",
997                                open_token.as_close_char(),
998                                close_token.as_close_char(),
999                            ),
1000                            span,
1001                        );
1002                    } else if open_list.is_empty() {
1003                        i.reset(&before);
1004                        return Ok(after);
1005                    }
1006                }
1007            }
1008
1009            let p = |i: &mut _| inner(i, open_token);
1010            let (checkpoint, inner) = p.with_taken().parse_next(i)?;
1011            i.reset(&checkpoint);
1012            Ok(Suffix::MacroCall(inner))
1013        }

1138        fn identifier_or_prefixed_string<'a: 'l, 'l>(
1139            i: &mut InputStream<'a, 'l>,
1140        ) -> ParseResult<'a, ()> {
1141            // <https://doc.rust-lang.org/reference/tokens.html#r-lex.token.literal.str-raw.syntax>
1142
1143            let ((prefix, hashes, quot), prefix_span): ((_, usize, _), _) =
1144                (identifier, repeat(.., '#'), opt('"'))
1145                    .with_span()
1146                    .parse_next(i)?;
1147            if hashes >= 256 {
1148                return cut_error!(
1149                    "a maximum of 255 hashes `#` are allowed with raw and prefixed strings",
1150                    prefix_span,
1151                );
1152            }
1153
1154            let str_kind = match prefix {
1155                // raw cstring or byte slice
1156                "br" => Some(StrPrefix::Binary),
1157                "cr" => Some(StrPrefix::CLike),
1158                // raw string string or identifier
1159                "r" => None,
1160                // a simple identifier
1161                _ if hashes == 0 && quot.is_none() => return Ok(()),
1162                // reserved prefix: reject
1163                _ => {
1164                    return cut_error!(
1165                        format!("reserved prefix `{}#`", prefix.escape_debug()),
1166                        prefix_span,
1167                    );
1168                }
1169            };
1170
1171            if quot.is_some() {
1172                // got a raw string
1173
1174                let delim = format!("\"{:#<hashes$}", "");
1175                let p = terminated(take_until(.., delim.as_str()).with_span(), delim.as_str());
1176                let Some((inner, inner_span)) = opt(p).parse_next(i)? else {
1177                    return cut_error!("unterminated raw string", prefix_span);
1178                };
1179
1180                if inner.split('\r').skip(1).any(|s| !s.starts_with('\n')) {
1181                    return cut_error!(
1182                        format!(
1183                            "a bare CR (Mac linebreak) is not allowed in string literals, \
1184                            use NL (Unix linebreak) or CRNL (Windows linebreak) instead, \
1185                            or type `\\r` explicitly",
1186                        ),
1187                        inner_span,
1188                    );
1189                }
1190
1191                let msg = match str_kind {
1192                    Some(StrPrefix::Binary) => inner
1193                        .bytes()
1194                        .any(|b| !b.is_ascii())
1195                        .then_some("binary string literals must not contain non-ASCII characters"),
1196                    Some(StrPrefix::CLike) => inner
1197                        .bytes()
1198                        .any(|b| b == 0)
1199                        .then_some("cstring literals must not contain NUL characters"),
1200                    None => None,
1201                };
1202                if let Some(msg) = msg {
1203                    return cut_error!(msg, prefix_span);
1204                }
1205
1206                not_suffix_with_hash(i)?;
1207                Ok(())
1208            } else if hashes == 0 {
1209                // a simple identifier
1210                Ok(())
1211            } else if let Some((id, span)) = opt(identifier.with_span()).parse_next(i)? {
1212                // got a raw identifier
1213
1214                if str_kind.is_some() {
1215                    // an invalid raw identifier like `cr#async`
1216                    cut_error!(
1217                        format!(
1218                            "reserved prefix `{}#`, only `r#` is allowed with raw identifiers",
1219                            prefix.escape_debug(),
1220                        ),
1221                        prefix_span,
1222                    )
1223                } else if hashes > 1 {
1224                    // an invalid raw identifier like `r##async`
1225                    cut_error!(
1226                        "only one `#` is allowed in raw identifier delimitation",
1227                        prefix_span,
1228                    )
1229                } else {
1230                    // a raw identifier like `r#async`
1231                    if matches!(id, "self" | "Self" | "super" | "crate" | "_") {
1232                        cut_error!(
1233                            format!("`{}` cannot be a raw identifier", id.escape_debug()),
1234                            span,
1235                        )

1262        fn punctuation<'a: 'l, 'l>(i: &mut InputStream<'a, 'l>) -> ParseResult<'a, ()> {
1263            // <https://doc.rust-lang.org/reference/tokens.html#punctuation>
1264            // hash '#' omitted
1265
1266            const ONE_CHAR: &[u8] = b"+-*/%^!&|=><@_.,;:$?~";
1267            const TWO_CHARS: &[[u8; 2]] = &[
1268                *b"&&", *b"||", *b"<<", *b">>", *b"+=", *b"-=", *b"*=", *b"/=", *b"%=", *b"^=",
1269                *b"&=", *b"|=", *b"==", *b"!=", *b">=", *b"<=", *b"..", *b"::", *b"->", *b"=>",
1270                *b"<-",
1271            ];
1272            const THREE_CHARS: &[[u8; 3]] = &[*b"<<=", *b">>=", *b"...", *b"..="];
1273
1274            let three_chars = take(3usize).verify_map(|head: &str| {
1275                if let Ok(head) = head.as_bytes().try_into()
1276                    && THREE_CHARS.contains(head)
1277                {
1278                    Some(())
1279                } else {
1280                    None
1281                }
1282            });
1283            let two_chars = take(2usize).verify_map(|head: &str| {
1284                if let Ok(head) = head.as_bytes().try_into()
1285                    && TWO_CHARS.contains(head)
1286                {
1287                    Some(())
1288                } else {
1289                    None
1290                }
1291            });
1292            let one_char = any.verify_map(|head: char| {
1293                if let Ok(head) = head.try_into()
1294                    && ONE_CHAR.contains(&head)
1295                {
1296                    Some(())
1297                } else {
1298                    None
1299                }
1300            });
1301
1302            // need to check long to short
1303            alt((three_chars, two_chars, one_char)).parse_next(i)
1304        }

1377    fn r#struct(i: &mut InputStream<'a, 'l>) -> ParseResult<'a, WithSpan<Self>> {
1378        let _level_guard = i.state.level.nest(i)?;
1379        let mut p = (
1380            ws('{'.span()),
1381            cut_err((separated(
1382                0..,
1383                alt((Self::struct_field, Self::struct_base)),
1384                ws(','),
1385            ),)),
1386            opt(ws(',')), // Trailing comma.
1387            opt(ws(winnow::token::any.with_span())),
1388        );
1389        let (span, (all_fields,), trailing_comma, closed): (
1390            _,
1391            (Vec<Field<'_>>,),
1392            Option<_>,
1393            Option<_>,
1394        ) = p.parse_next(i)?;
1395        if trailing_comma.is_some() && all_fields.is_empty() {
1396            return cut_error!("missing field before `,`", span);
1397        }
1398        let mut base: Option<WithSpan<Box<Expr<'a>>>> = None;
1399        let mut fields = Vec::with_capacity(all_fields.len());
1400        for field in all_fields {
1401            match field {
1402                Field::Field(field) => {
1403                    if base.is_some() {
1404                        return cut_error!(
1405                            "expected end of struct expression after `..` was used",
1406                            field.name.span()
1407                        );
1408                    }
1409                    fields.push(field);
1410                }
1411                Field::Base(new_base) => {
1412                    if base.is_some() {
1413                        return cut_error!(
1414                            "expected end of struct expression after `..` was used",
1415                            new_base.span()
1416                        );
1417                    }
1418                    base = Some(new_base);
1419                }
1420            }
1421        }
1422        if closed.as_ref().is_none_or(|(c, _)| *c != '}') {
1423            let err_span = match closed {
1424                Some((_, span)) => span,
1425                _ => span,
1426            };
1427            if base.is_some() {
1428                return cut_error!(
1429                    "expected end of struct expression after `..` was used",
1430                    err_span
1431                );
1432            } else if !fields.is_empty() {
1433                return cut_error!("expected `,`, `..`, field name or `}`", err_span);
1434            } else {
1435                return cut_error!("expected field name, `..` or `}`", err_span);
1436            }
1437        }
1438
1439        Ok(WithSpan::new(Self::Struct { fields, base }, span))
1440    }

1442    fn struct_base(i: &mut InputStream<'a, 'l>) -> ParseResult<'a, Field<'a>> {
1443        let ((_, base_expr), span) = (ws(".."), opt(ws(move |i: &mut _| Expr::parse(i, true))))
1444            .with_span()
1445            .parse_next(i)?;
1446        match base_expr {
1447            Some(base_expr) => Ok(Field::Base(base_expr)),
1448            None => cut_error!("expected expression after `..`", span),
1449        }
1450    }

1452    fn struct_field(i: &mut InputStream<'a, 'l>) -> ParseResult<'a, Field<'a>> {
1453        let ((name, name_span), has_colon, value) = alt((
1454            (
1455                alt((identifier, digit1)).with_span(),
1456                ws(':'),
1457                opt(ws(|i: &mut _| Expr::parse(i, true))),
1458            )
1459                .map(|(name, _, expr)| (name, true, expr)),
1460            identifier.with_span().map(|name| (name, false, None)),
1461        ))
1462        .parse_next(i)?;
1463        if has_colon && value.is_none() {
1464            cut_error!("expected expression after `:`", *i)
1465        } else {
1466            Ok(Field::Field(ExprStructField {
1467                name: WithSpan::new(name, name_span),
1468                value,
1469            }))
1470        }
1471    }

1538#[derive(Clone, Debug, PartialEq)]
1539pub enum TyGenericsKind<'a> {
1540    Path {
1541        path: Vec<WithSpan<&'a str>>,
1542        args: Option<WithSpan<Vec<WithSpan<TyGenerics<'a>>>>>,
1543    },
1544    Tuple(Vec<WithSpan<TyGenerics<'a>>>),
1545    Array {
1546        ty: Box<WithSpan<TyGenerics<'a>>>,
1547        nb_elems: Option<&'a str>,
1548    },
1549}
1550
1551impl<'a: 'l, 'l> TyGenericsKind<'a> {
1552    fn parse(i: &mut InputStream<'a, 'l>) -> ParseResult<'a, Self> {
1553        let _level_guard = i.state.level.nest(i)?;
1554        alt((Self::tuple, Self::array, Self::ty_path)).parse_next(i)
1555    }
1556
1557    fn ty_path(i: &mut InputStream<'a, 'l>) -> ParseResult<'a, TyGenericsKind<'a>> {
1558        let path = separated(
1559            1..,
1560            ws(identifier
1561                .with_span()
1562                .map(|(name, span)| WithSpan::new(name, span))),
1563            "::",
1564        )
1565        .map(|v: Vec<_>| v);
1566
1567        let (path, args) = (path, opt(TyGenerics::args)).parse_next(i)?;
1568
1569        if let [name] = path.as_slice() {
1570            if matches!(**name, "super" | "self" | "crate") {
1571                // `Self` and `_` are allowed
1572                return err_reserved_identifier(name);
1573            }
1574        } else {
1575            for (idx, name) in path.iter().enumerate() {
1576                if **name == "_" {
1577                    // `_` is never allowed
1578                    return err_underscore_identifier(name);
1579                } else if idx > 0 && matches!(**name, "super" | "self" | "Self" | "crate") {
1580                    // At the front of the path, "super" | "self" | "Self" | "crate" are allowed.
1581                    // Inside the path, they are not allowed.
1582                    return err_reserved_identifier(name);
1583                }
1584            }
1585        }
1586        Ok(TyGenericsKind::Path { path, args })
1587    }
1588
1589    fn tuple(i: &mut InputStream<'a, 'l>) -> ParseResult<'a, TyGenericsKind<'a>> {
1590        let start = *i;
1591        // We ensure we're in the right function to get better errors later on.
1592        ws('(').parse_next(i)?;
1593        let Ok(tuple_elems) = separated(0.., TyGenerics::parse, ws(',')).parse_next(i) else {
1594            return cut_error!("expected a list of type separated by a comma", start);
1595        };
1596        if (opt(ws(',')), ws(')')).parse_next(i).is_err() {
1597            return cut_error!("expected a list of type separated by a comma", start);
1598        }
1599        Ok(TyGenericsKind::Tuple(tuple_elems))
1600    }
1601
1602    fn array(i: &mut InputStream<'a, 'l>) -> ParseResult<'a, TyGenericsKind<'a>> {
1603        let start = *i;
1604        // We ensure we're in the right function to get better errors later on.
1605        ws('[').parse_next(i)?;
1606
1607        let ty = match TyGenerics::parse.parse_next(i) {
1608            Ok(ty) => ty,
1609            Err(error @ ErrMode::Cut(_)) => return Err(error),
1610            Err(_) => return cut_error!("expected a type", *i),
1611        };
1612        let mut nb_elems = None;
1613        if let Ok((_, colon_span)) = ws(';').with_span().parse_next(i) {
1614            let Ok((parsed_nb, nb_span)) = num_lit.with_span().parse_next(i) else {
1615                return cut_error!("expected a number after `;`", colon_span);
1616            };
1617            match parsed_nb {
1618                Num::Int(nb, Some(crate::IntKind::Usize) | None) => nb_elems = Some(nb),
1619                Num::Int(_, Some(kind)) => {
1620                    return cut_error!(
1621                        format!("array size should be `usize`, found `{kind}`"),
1622                        nb_span,
1623                    );
1624                }
1625                Num::Float(nb, _) => {
1626                    return cut_error!(
1627                        format!("expected a number after `;`, found a float (`{nb}`)"),
1628                        nb_span,
1629                    );
1630                }
1631            }
1632        }
1633        if ws(']').parse_next(i).is_err() {
1634            return cut_error!("missing `]` to close the array", start);
1635        }
1636        Ok(TyGenericsKind::Array {
1637            ty: Box::new(ty),
1638            nb_elems,
1639        })
1640    }
1641}
