1//! askama::filter_fn - Proc macro implementation
2//! This parses an annotated freestanding function annotated with the `filter_fn` attribute
3//! into an internal intermediate representation (the `FilterSignature` struct).
4//! Then, the output code is generated from said struct.
5

20pub fn derive_filter_fn(
21    attr: TokenStream,
22    item: TokenStream,
23    import_askama: fn() -> TokenStream,
24) -> TokenStream {
25    let ffn: ItemFn = match parse_ts_or_compile_error(item, import_askama) {
26        ControlFlow::Continue(ffn) => ffn,
27        ControlFlow::Break(err) => return err,
28    };
29    match filter_fn_impl(attr, &ffn) {
30        Ok(tt) => tt,
31        Err(CompileError { msg, span }) => {
32            let import_askama = import_askama();
33            quote_spanned! {
34                span.unwrap_or_else(|| ffn.sig.ident.span()) =>
35                const _: () = {
36                    #import_askama
37                    askama::helpers::core::compile_error!(#msg);
38                };
39            }
40        }
41    }
42}

44/// Helper macro to produce proc macro compiler error messages with a given span
45/// if a given condition is not met.
46macro_rules! p_assert {
47    ($cond:expr, $span:expr => $msg:literal $(,)?) => {
48        match $cond {
49            true => Ok(()),
50            false => p_err!($span => $msg)
51        }
52    };
53}
54
55macro_rules! p_err {
56    ($span:expr => $msg:literal $(,)?) => {
57        Err(CompileError::new_with_span_stable($msg, None, Some($span)))
58    };
59}

63/// Internal representation for custom required filter arguments
64struct FilterArgumentRequired {
65    idx: usize,
66    ident: Ident,
67    mutability: Option<Mut>,
68    ty: Type,
69    generics: HashSet<Ident>,
70}
71
72/// Internal representation for custom optional filter arguments.
73/// These are arguments for which an optional value was supplied using `#[optional(...)]`
74struct FilterArgumentOptional {
75    idx: usize,
76    ident: Ident,
77    mutability: Option<Mut>,
78    ty: Type,
79    default: Expr,
80}

82/// Internal representation for a filter function's lifetime.
83#[derive(Clone)]
84struct FilterLifetime {
85    lifetime: Lifetime,
86    bounds: Punctuated<Lifetime, Token![+]>,
87    used_by_extra_args: bool,
88}
89
90/// Internal representation for a filter function's generic argument.
91#[derive(Clone)]
92struct FilterArgumentGeneric {
93    ident: Ident,
94    bounds: Punctuated<TypeParamBound, Token![+]>,
95}

97fn get_lifetimes(stream: TokenStream, lifetimes: &mut HashSet<Ident>) {
98    let mut iterator = stream.into_iter().peekable();
99    while let Some(token) = iterator.next() {
100        match token {
101            TokenTree::Group(g) => get_lifetimes(g.stream(), lifetimes),
102            TokenTree::Punct(p) if p.as_char() == '\'' => {
103                // Lifetimes are represented as `[Punct('), Ident("a")]` in the `TokenStream`.
104                if let Some(TokenTree::Ident(i)) = iterator.peek() {
105                    lifetimes.insert(i.clone());
106                }
107            }
108            TokenTree::Punct(_) | TokenTree::Ident(_) | TokenTree::Literal(_) => continue,
109        }
110    }
111}

113/// A freestanding method annotated with `askama::filter_fn` is parsed into an instance of this
114/// struct, and then the resulting code is generated from there.
115/// This struct serves as an intermediate representation after some preprocessing on the raw AST.
116struct FilterSignature {
117    /// Name of the annotated freestanding filter function
118    ident: Ident,
119    /// Lifetime bounds.
120    lifetimes: Vec<FilterLifetime>,
121    /// Name of the input variable
122    arg_input: FilterArgumentRequired,
123    /// Name of the askama environment variable
124    arg_env: FilterArgumentRequired,
125    /// Generic parameters required for the filter's input argument
126    arg_input_generics: Vec<FilterArgumentGeneric>,
127    /// Required filter arguments
128    args_required: Vec<FilterArgumentRequired>,
129    /// Optional filter arguments - each of which has a default value
130    args_optional: Vec<FilterArgumentOptional>,
131    /// Generic parameters in use by the required filter arguments
132    args_required_generics: HashMap<Ident, FilterArgumentGeneric>,
133    /// The `where` clause of the source function
134    where_clause: Option<WhereClause>,
135    /// Filter function result type
136    result_ty: ReturnType,
137}

142impl FilterSignature {
143    /// Parse the annotated function's signature and do some preprocessing to produce an instance
144    /// of the `FilterSignature` struct.
145    fn try_from_signature(sig: &Signature) -> Result<FilterSignature, CompileError> {
146        // preliminary validation
147        p_assert!(!sig.inputs.is_empty(), sig.paren_token.span.open() =>
148            "Filter function missing required input and environment arguments. Example: \
149            `fn filter0(_: &dyn std::fmt::Display, _: &dyn askama::Values) -> askama::Result<String>`"
150        )?;
151        p_assert!(sig.inputs.len() >= 2, sig.paren_token.span.open() =>
152            "Filter function missing required environment argument. Example: \
153            `fn filter0(_: &dyn std::fmt::Display, _: &dyn askama::Values) -> askama::Result<String>`"
154        )?;
155        if let Some(gc_arg) = sig.generics.const_params().next() {
156            p_err!(gc_arg.span() => "Const generics are currently not supported for filters")?;
157        }
158        p_assert!(
159            matches!(sig.output, ReturnType::Type(_, _)),
160            sig.paren_token.span.close() => "Filter function is missing return type"
161        )?;
162
163        // ########################################
164        // generics
165        let mut generics = HashMap::default();
166        for gp in sig.generics.type_params() {
167            p_assert!(gp.default.is_none(), gp.default.span() => "Filter functions don't support generic parameter defaults")?;
168
169            let ident = gp.ident.clone();
170            let bounds = gp.bounds.clone();
171            generics.insert(ident.clone(), FilterArgumentGeneric { ident, bounds });
172        }
173
174        // ########################################
175        // fixed arguments (input & env)
176        let arg_input = Self::try_get_fixed_arg(&sig.inputs[0], &generics)?;
177        let arg_input_generics: Vec<_> = arg_input
178            .generics
179            .iter()
180            .map(|i| generics[i].clone())
181            .collect();
182        let arg_env = Self::try_get_fixed_arg(&sig.inputs[1], &generics)?;
183
184        // ########################################
185        // user arguments
186        let mut args_required = vec![];
187        let mut args_optional = vec![];
188        let mut args_required_generics = HashMap::default();
189        let mut lifetimes_used_in_non_required = HashSet::default();
190        for (arg_idx, arg) in sig.inputs.iter().skip(2).enumerate() {
191            let FnArg::Typed(arg) = arg else {
192                continue;
193            };
194            let Pat::Ident(arg_pat) = &*arg.pat else {
195                p_err!(arg.pat.span() => "Only conventional function arguments are supported")?
196            };
197            p_assert!(
198                !matches!(*arg.ty, Type::ImplTrait(_)),
199                arg.ty.span() => "Impl generics are currently not supported for filters"
200            )?;
201            get_lifetimes(arg.to_token_stream(), &mut lifetimes_used_in_non_required);
202
203            // reference-parameters without explicit lifetime, inherit the 'filter lifetime
204            let arg_type = patch_ref_with_lifetime(&arg.ty, &format_ident!("filter"));
205
206            match Self::get_optional_arg_attr(arg)? {
207                // required argument (= has no default value)
208                None => {
209                    // required argument
210                    p_assert!(args_optional.is_empty(), arg.span() => "All required arguments must appear before any optional ones")?;
211                    // determine all generic parameters used by this argument
212                    let used_generics: HashSet<_> = generics
213                        .keys()
214                        .filter(|i| type_contains_ident(&arg.ty, i).is_some())
215                        .cloned()
216                        .collect();
217                    // mark the used generic parameters
218                    used_generics.iter().map(|i| &generics[i]).for_each(|g| {
219                        args_required_generics.insert(g.ident.clone(), g.clone());
220                    });
221                    args_required.push(FilterArgumentRequired {
222                        idx: arg_idx,
223                        ident: arg_pat.ident.clone(),
224                        mutability: arg_pat.mutability,
225                        ty: arg_type,
226                        generics: used_generics,
227                    });
228                }
229                // optional argument (= has default value)
230                Some(default) => {
231                    // check if the argument uses any generics (which is not allowed for optional arguments)
232                    if let Some(span) = generics
233                        .keys()
234                        .filter_map(|i| type_contains_ident(&arg.ty, i))
235                        .next()
236                    {
237                        p_err!(span => "Optional arguments must not use generic parameters")?;
238                    }
239
240                    args_optional.push(FilterArgumentOptional {
241                        idx: arg_idx,
242                        ident: arg_pat.ident.clone(),
243                        mutability: arg_pat.mutability,
244                        ty: arg_type,
245                        default,
246                    });
247                }
248            }
249        }
250        // lifetimes
251        let lifetimes = sig
252            .generics
253            .lifetimes()
254            .map(|lt| {
255                let lifetime = lt.lifetime.clone();
256                let bounds = lt.bounds.clone();
257                let used_by_extra_args = lifetimes_used_in_non_required.contains(&lifetime.ident);
258                FilterLifetime {
259                    lifetime,
260                    bounds,
261                    used_by_extra_args,
262                }
263            })
264            .collect::<Vec<_>>();
265
266        // ########################################
267
268        Ok(FilterSignature {
269            ident: sig.ident.clone(),
270            lifetimes,
271            arg_input,
272            arg_input_generics,
273            arg_env,
274            args_required,
275            args_optional,
276            args_required_generics,
277            where_clause: sig.generics.where_clause.clone(),
278            result_ty: sig.output.clone(),
279        })
280    }
281
282    /// Parse one of the fixed filter arguments (value and env)
283    fn try_get_fixed_arg(
284        arg: &FnArg,
285        generics: &HashMap<Ident, FilterArgumentGeneric>,
286    ) -> Result<FilterArgumentRequired, CompileError> {
287        let FnArg::Typed(arg) = arg else {
288            p_err!(arg.span() => "Illegal or unsupported type of argument for filter function")?
289        };
290        let (arg_ident, mutability) = match &*arg.pat {
291            Pat::Ident(pat_ident) => (pat_ident.ident.clone(), pat_ident.mutability),
292            Pat::Wild(pat) => (Ident::new("_", pat.span()), None), // little hack
293            _ => p_err!(arg.pat.span() => "Only conventional function arguments are supported")?,
294        };
295
296        Ok(FilterArgumentRequired {
297            idx: 0,
298            ident: arg_ident,
299            ty: *arg.ty.clone(),
300            mutability,
301            generics: generics
302                .keys()
303                .filter(|i| type_contains_ident(&arg.ty, i).is_some())
304                .cloned()
305                .collect(),
306        })
307    }
308
309    /// Parse the `#[optional(<default_expr>)]` attribute found on filter function arguments.
310    /// For optional arguments, this is mandatory.
311    fn get_optional_arg_attr(arg: &PatType) -> Result<Option<Expr>, CompileError> {
312        for attr in &arg.attrs {
313            if let Some(ident) = attr.meta.path().get_ident()
314                && ident == "optional"
315            {
316                let default: Expr = match attr.parse_args() {
317                    Ok(default) => default,
318                    Err(_) => p_err!(attr.span() => "Default argument not a valid expression")?,
319                };
320                return Ok(Some(default));
321            }
322        }
323        Ok(None)
324    }
325}

330impl FilterSignature {
331    /// Returns a tuple containing two items:
332    ///
333    /// 1. The list of lifetimes with their bounds.
334    /// 2. The list of lifetimes without their bounds.
335    fn lifetimes_bounds<F: Fn(&FilterLifetime) -> bool>(
336        &self,
337        filter: F,
338    ) -> (Vec<TokenStream>, Vec<&Lifetime>) {
339        let mut lifetimes = Vec::with_capacity(self.lifetimes.len());
340        let mut lifetimes_no_bounds = Vec::with_capacity(self.lifetimes.len());
341        for lt in &self.lifetimes {
342            if !filter(lt) {
343                continue;
344            }
345            let name = &lt.lifetime;
346            let bounds = &lt.bounds;
347            lifetimes.push(quote! { #name: #bounds });
348            lifetimes_no_bounds.push(name);
349        }
350        (lifetimes, lifetimes_no_bounds)
351    }
352
353    fn lifetimes_fillers<F: Fn(&FilterLifetime) -> bool>(&self, filter: F) -> Vec<TokenStream> {
354        self.lifetimes
355            .iter()
356            .filter(|l| filter(l))
357            .map(|_| quote! { '_ })
358            .collect()
359    }
360
361    /// Generates a struct named after the filter function.
362    /// This struct will contain all the filter's arguments (except input and env).
363    /// The struct is basically a builder pattern for the custom filter arguments.
364    /// It is structured like this:
365    /// - All required arguments (no default value supplied) are contained in an `Option`
366    /// - All optional arguments (default value supplied via attr) are contained as is
367    /// - The struct adopts all the generic parameters which are in use by custom args
368    /// - The struct always has a 'filter lifetime. This is the default lifetime relating
369    ///   to the struct instance itself. It is patched onto reference arguments without
370    ///   own explicit lifetime.
371    /// - The struct has one const generic bool parameter for each required argument, tracking
372    ///   whether the required argument was supplied.
373    ///
374    /// For every user argument to the filter, we implement the trait:
375    /// `askama::filters::ValidArgIdx<const IDX: usize>` on the generated struct (where IDX = arg.idx).
376    /// During code generation, the line: `const _: bool = askama::filters::ValidArgIdx<n>::VALID`
377    /// can then check at compile-time whether there is an argument with the given index.
378    fn gen_struct_definition(&self, vis: &Visibility) -> TokenStream {
379        let ident = &self.ident;
380        // struct generic parameters
381        let struct_generics = self
382            .args_required_generics
383            .values()
384            .map(|g| g.ident.clone());
385        let required_flags = self
386            .args_required
387            .iter()
388            .map(|a| format_ident!("REQUIRED_ARG_FLAG_{}", a.idx));
389        // struct field definitions
390        let required_fields = self.args_required.iter().map(|arg| {
391            let (name, ty) = (&arg.ident, &arg.ty);
392            quote! { #name: Option<#ty> }
393        });
394        let optional_fields = self.args_optional.iter().map(|arg| {
395            let (name, ty) = (&arg.ident, &arg.ty);
396            quote! { #name: #ty }
397        });
398        // introspection (better compile error messages on misuse)
399        let required_arg_cnt = self.args_required.len();
400        let optional_arg_cnt = self.args_optional.len();
401        let arg_cnt = required_arg_cnt + optional_arg_cnt;
402        let lifetimes_fillers = self.lifetimes_fillers(|l| l.used_by_extra_args);
403        let valid_arg_impls = (0..arg_cnt).map(|idx| {
404            quote! {
405                #[diagnostic::do_not_recommend]
406                impl askama::filters::ValidArgIdx<#idx> for #ident<'_, #(#lifetimes_fillers,)*> {}
407            }
408        });
409
410        let (_, lifetimes) = self.lifetimes_bounds(|l| l.used_by_extra_args);
411        quote! {
412            #[allow(non_camel_case_types)]
413            #vis struct #ident<'filter, #(#lifetimes,)* #(#struct_generics = (),)* #(const #required_flags : bool = false,)*> {
414                _lifetime: std::marker::PhantomData<&'filter ()>,
415                /* required fields */
416                #(#required_fields,)*
417                /* optional fields */
418                #(#optional_fields,)*
419            }
420
421            #(#valid_arg_impls)*
422        }
423    }
424
425    /// Generate the entry-point for the filter builder struct.
426    /// This fills all required arguments (`Option<Ty>`) as `None`, and all the optional
427    /// arguments with the default value supplied in the `#[optional(..)]` attribute.
428    ///
429    /// This entry point starts with a type of `()` for all the generic parameter
430    /// used by required arguments. They are only filled with the correct type
431    /// as soon as the argument is supplied into the corresponding setter.
432    fn gen_default_impl(&self) -> TokenStream {
433        let ident = &self.ident;
434        // initial field values
435        let required_defaults = self
436            .args_required
437            .iter()
438            .map(|a| &a.ident)
439            .map(|i| quote! { #i: None });
440        let optional_defaults = self.args_optional.iter().map(|a| {
441            let ident = &a.ident;
442            let value = &a.default;
443            quote! { #ident: #value }
444        });
445        let lifetimes_fillers = self.lifetimes_fillers(|l| l.used_by_extra_args);
446
447        quote! {
448            impl std::default::Default for #ident<'_, #(#lifetimes_fillers,)*> {
449                fn default() -> Self {
450                    Self {
451                        _lifetime: std::marker::PhantomData::default(),
452                        #(#required_defaults,)*
453                        #(#optional_defaults,)*
454                    }
455                }
456            }
457        }
458    }
459
460    /// Generate the builder-style setter methods. Each argument gets two methods.
461    /// - with_{arg_name}() for when an argument is passed as named arg
462    /// - with_{arg_idx}() for when an argument is passed as positional arg
463    ///
464    /// Positional setters are always implemented by calling their named counterparts.
465    ///
466    /// Since optional arguments don't support generic arguments and their presence
467    /// does not need to be checked, they simply change the field's value in the struct.
468    /// Whereas required arguments construct a new struct instance, because they need to
469    /// - Patch generic arguments (that started out with `()`)
470    /// - Change the const generic bool parameter that tracks their presence to `true`
471    fn gen_setters(&self) -> TokenStream {
472        let optional_setters = self.gen_setters_optional();
473        let required_setters = self
474            .args_required
475            .iter()
476            .map(|arg| self.gen_required_setter(arg));
477
478        quote! {
479            #optional_setters
480            #(#required_setters)*
481        }
482    }
483
484    /// This generates setters for required arguments - which is much more complex than
485    /// optional arguments. Each setter for a required argument:
486    /// - constructs a new instance of the builder struct.
487    /// - patches the required arguments' tracking const bool flag to `true`
488    /// - fills the required argument's corresponding generic type arguments
489    ///
490    /// So setters for required arguments do not just return a copy of the builder struct,
491    /// they also change its type signature (due to differing generic arguments).
492    fn gen_required_setter(&self, arg: &FilterArgumentRequired) -> TokenStream {
493        let ident = &self.ident;
494        let cur_arg_ident = &arg.ident;
495        let cur_arg_ty = &arg.ty;
496        // setter idents
497        let named_ident = format_ident!("with_{}", arg.ident);
498        let positional_ident = format_ident!("with_{}", arg.idx);
499        // impl generics
500        let required_generics_impl: Vec<_> = self
501            .args_required_generics
502            .keys()
503            .map(|i| format_ident!("{}__OLD", i))
504            .collect();
505        let required_flags: Vec<_> = self
506            .args_required
507            .iter()
508            .map(|a| format_ident!("REQUIRED_ARG_FLAG_{}", a.idx))
509            .collect();
510        // function generics
511        let required_generics_fn: Vec<_> = arg
512            .generics
513            .iter()
514            .map(|i| &self.args_required_generics[i])
515            .map(|g| {
516                let ident = &g.ident;
517                let bounds = &g.bounds;
518                quote! { #ident: #bounds }
519            })
520            .collect();
521        let (_, lifetimes_no_bounds) = self.lifetimes_bounds(|l| l.used_by_extra_args);
522        // return type
523        let fn_return_ty = {
524            let required_generics_result =
525                self.args_required_generics
526                    .keys()
527                    .map(|i| match arg.generics.contains(i) {
528                        true => i.clone(),
529                        false => format_ident!("{}__OLD", i),
530                    });
531            let required_flags_result = self.args_required.iter().map(|a| {
532                match a.idx == arg.idx {
533                    true => quote!(true), // current arg
534                    false => format_ident!("REQUIRED_ARG_FLAG_{}", a.idx).to_token_stream(),
535                }
536            });
537            quote! { #ident<'filter, #(#lifetimes_no_bounds,)* #(#required_generics_result,)* #(#required_flags_result,)*> }
538        };
539        // struct fields - (all fields, except that of current argument)
540        let other_required_fields = self
541            .args_required
542            .iter()
543            .filter(|a| a.idx != arg.idx)
544            .map(|a| &a.ident)
545            .map(|i| quote! { #i: self.#i });
546        let optional_fields = self.args_optional.iter().map(|a| &a.ident);
547
548        quote! {
549            #[allow(non_camel_case_types)]
550            impl<'filter, #(#lifetimes_no_bounds,)* #(#required_generics_impl,)* #(const #required_flags: bool,)*>
551            #ident<'filter, #(#lifetimes_no_bounds,)* #(#required_generics_impl,)* #(#required_flags,)*> {
552                // named setter
553                #[inline(always)]
554                pub fn #named_ident<#(#required_generics_fn,)*>(self, new_value: #cur_arg_ty) -> #fn_return_ty {
555                    // construct new instance of filter builder struct, by copying over all current values.
556                    // But replace the value of the setter's corresponding field with `Some(new_value)`.
557                    #ident {
558                        _lifetime: self._lifetime,
559                        // copy previous field values (all except field of current setter)
560                        #(#other_required_fields,)*
561                        #(#optional_fields: self.#optional_fields,)*
562                        // patch field of current argument to new value
563                        #cur_arg_ident: Some(new_value)
564                    }
565                }
566
567                // positional setter
568                #[inline(always)]
569                pub fn #positional_ident<#(#required_generics_fn,)*>(self, new_value: #cur_arg_ty) -> #fn_return_ty {
570                    self.#named_ident(new_value)
571                }
572            }
573        }
574    }

576    /// Generate setters for optional arguments
577    /// Compared to required arguments, they don't need to create a new struct instance,
578    /// because they don't need to change the struct's generic parameters.
579    /// Each getter just overwrites its corresponding field with the new value.
580    fn gen_setters_optional(&self) -> TokenStream {
581        let ident = &self.ident;
582        // generics (use stupid enumeration instead of named arguments for simplicity)
583        let required_generics: Vec<_> = (0..self.args_required_generics.len())
584            .map(|i| format_ident!("T{}", i))
585            .collect();
586        let required_flags: Vec<_> = (0..self.args_required.len())
587            .map(|i| format_ident!("F{}", i))
588            .collect();
589
590        let optional_setters = self.args_optional.iter().map(|arg| {
591            let arg_ident = &arg.ident;
592            let named_ident = format_ident!("with_{arg_ident}");
593            let positioned_ident = format_ident!("with_{}", arg.idx);
594            let arg_ty = &arg.ty;
595
596            quote! {
597                // named setter
598                #[inline(always)]
599                pub fn #named_ident(mut self, value: #arg_ty) -> Self {
600                    self.#arg_ident = value;
601                    self
602                }
603                // positional setter
604                #[inline(always)]
605                pub fn #positioned_ident(self, value: #arg_ty) -> Self {
606                    self.#named_ident(value)
607                }
608            }
609        });
610
611        let (_, lifetimes_no_bounds) = self.lifetimes_bounds(|l| l.used_by_extra_args);
612        quote! {
613            #[allow(non_camel_case_types)]
614            impl<'filter, #(#lifetimes_no_bounds,)* #(#required_generics,)* #(const #required_flags: bool,)*>
615            #ident<'filter, #(#lifetimes_no_bounds,)* #(#required_generics,)* #(#required_flags,)*> {
616                #(#optional_setters)*
617            }
618        }
619    }

621    /// Generate the `execute(input, env)` method that does the filter's actual work.
622    /// This method only takes the filter's input (before the pipe), as well as the askama values
623    /// environment variable as arguments.
624    /// The method is contained in an `impl{}` block that guards against missing required arguments
625    /// by requiring the value `true` for all argument-tracking const generic bool parameters.
626    ///
627    /// The method's body contains an internal "preamble" that first maps all of the struct's
628    /// fields into the local context by consuming them. Required arguments are unwrapped from
629    /// their `Option<>` container, and optional arguments are moved as is.
630    /// Then, the actual filter code is inserted after.
631    fn gen_exec_impl(&self, sig: &Signature, filter_impl: &Block) -> TokenStream {
632        let ident = &self.ident;
633        // input variable
634        // method generics (only the parameters not already present on struct)
635        let input_ident = &self.arg_input.ident;
636        let input_mutability = &self.arg_input.mutability;
637        let input_ty = &self.arg_input.ty;
638        let input_bounds = self
639            .arg_input_generics
640            .iter()
641            .filter(|g| !self.args_required_generics.contains_key(&g.ident))
642            .map(|g| {
643                let ident = &g.ident;
644                let bounds = &g.bounds;
645                quote! { #ident: #bounds }
646            });
647        let (all_lifetimes, _) = self.lifetimes_bounds(|_| true);
648        let (_, type_lifetimes) = self.lifetimes_bounds(|l| l.used_by_extra_args);
649
650        // env variable
651        let env_ident = &self.arg_env.ident;
652        let env_ty = &self.arg_env.ty;
653
654        // struct generics
655        let required_generics: Vec<_> = self
656            .args_required_generics
657            .values()
658            .map(|g| &g.ident)
659            .collect();
660        let required_generic_bounds = self.args_required_generics.values().map(|g| &g.bounds);
661        let required_flags = std::iter::repeat_n(quote!(true), self.args_required.len());
662
663        // filter result
664        let result_ty = &self.result_ty;
665
666        // variables
667        let required_args = self.args_required.iter().map(|a| {
668            let mutability = a.mutability;
669            let ident = &a.ident;
670            quote! {
671                let #mutability #ident = unsafe { self.#ident.unwrap_unchecked() };
672            }
673        });
674        let optional_args = self.args_optional.iter().map(|a| {
675            let mutability = a.mutability;
676            let ident = &a.ident;
677            quote! {
678                let #mutability #ident = unsafe { self.#ident };
679            }
680        });
681
682        let fn_token = &sig.fn_token;
683        let where_clause = self.where_clause.as_ref();
684        let impl_generics = quote! { #(#required_generics: #required_generic_bounds,)* };
685        let impl_struct_generics = quote! { #(#required_generics,)* #(#required_flags,)* };
686        let lifetimes_fillers = self.lifetimes_fillers(|l| l.used_by_extra_args);
687        quote_spanned! {
688            sig.paren_token.span =>
689            // if all required arguments have been supplied (P0 == true, P1 == true)
690            // ... the execute() method is "unlocked":
691            impl<#(#all_lifetimes,)* #impl_generics> #ident<'_, #(#type_lifetimes,)* #impl_struct_generics> {
692                #[inline(always)]
693                pub #fn_token execute< #(#input_bounds,)* >(
694                    self,
695                    #input_mutability #input_ident: #input_ty,
696                    #env_ident: #env_ty
697                ) #result_ty #where_clause {
698                    // map filter variables with original name into scope
699                    #( #required_args )*
700                    #( #optional_args )*
701                    // insert actual filter function implementation
702                    #filter_impl
703                }
704            }
705
706            impl<#impl_generics> askama::filters::ValidFilterInvocation for #ident<'_, #(#lifetimes_fillers,)* #impl_struct_generics> {}
707        }
708    }
709}

713fn filter_fn_impl(attr: TokenStream, ffn: &ItemFn) -> Result<TokenStream, CompileError> {
714    p_assert!(
715        attr.is_empty(),
716        attr.span() => "`#[askama::filter_fn]` does not expect any attributes"
717    )?;
718
719    let fsig = FilterSignature::try_from_signature(&ffn.sig)?;
720
721    for gp in &ffn.sig.generics.params {
722        match gp {
723            GenericParam::Type(_) | GenericParam::Lifetime(_) => {}
724            GenericParam::Const(_) => {
725                p_err!(gp.span() => "Const generic arguments are not supported for now")?;
726            }
727        }
728    }
729
730    let struct_def = fsig.gen_struct_definition(&ffn.vis);
731    let default_impl = fsig.gen_default_impl();
732    let setter_impl = fsig.gen_setters();
733    let exec_impl = fsig.gen_exec_impl(&ffn.sig, &ffn.block);
734
735    Ok(quote!(
736        #struct_def
737        #default_impl
738        #setter_impl
739        #exec_impl
740    ))
741}

743/// Recursively check if a type contains one of the given Idents
744fn type_contains_ident(ty: &Type, ident: &Ident) -> Option<Span> {
745    match ty {
746        Type::Path(type_path) => {
747            for segment in &type_path.path.segments {
748                // Check if the segment ident matches
749                if &segment.ident == ident {
750                    return Some(segment.ident.span());
751                }
752
753                // Check generic arguments recursively
754                if let syn::PathArguments::AngleBracketed(ref args) = segment.arguments {
755                    for arg in &args.args {
756                        match arg {
757                            syn::GenericArgument::Type(inner_ty) => {
758                                if let Some(span) = type_contains_ident(inner_ty, ident) {
759                                    return Some(span);
760                                }
761                            }
762                            syn::GenericArgument::AssocType(assoc) => {
763                                if let Some(span) = type_contains_ident(&assoc.ty, ident) {
764                                    return Some(span);
765                                }
766                            }
767                            _ => {} // Not types -> skip
768                        }
769                    }
770                }
771            }
772            None
773        }
774        Type::Reference(type_ref) => type_contains_ident(&type_ref.elem, ident),
775        Type::Slice(type_slice) => type_contains_ident(&type_slice.elem, ident),
776        Type::Array(type_array) => type_contains_ident(&type_array.elem, ident),
777        Type::Tuple(type_tuple) => type_tuple
778            .elems
779            .iter()
780            .filter_map(|elem_ty| type_contains_ident(elem_ty, ident))
781            .next(),
782        Type::Paren(type_paren) => type_contains_ident(&type_paren.elem, ident),
783        Type::Group(type_group) => type_contains_ident(&type_group.elem, ident),
784        _ => None, // covers everything else
785    }
786}

788fn patch_ref_with_lifetime(ty: &Type, lifetime: &Ident) -> Type {
789    match ty {
790        Type::Reference(type_ref) => {
791            let mut new_type_ref = type_ref.clone();
792
793            // Inject the lifetime if it's missing
794            if new_type_ref.lifetime.is_none() {
795                new_type_ref.lifetime = Some(Lifetime {
796                    apostrophe: Span::call_site(),
797                    ident: lifetime.clone(),
798                });
799            }
800
801            Type::Reference(new_type_ref)
802        }
803        _ => ty.clone(), // Only patch reference types; others remain unchanged
804    }
