35    /// Extract the template metadata from the `DeriveInput` structure. This
36    /// mostly recovers the data for the `TemplateInput` fields from the
37    /// `template()` attribute list fields.
38    pub(crate) fn new<'n>(
39        ast: &'n syn::DeriveInput,
40        enum_ast: Option<&'n syn::DeriveInput>,
41        config: &'n Config,
42        args: &'n TemplateArgs,
43    ) -> Result<TemplateInput<'n>, CompileError> {
44        let TemplateArgs {
45            template_span,
46            source: (source, source_span),
47            block,
48            blocks,
49            print,
50            escaping,
51            ext,
52            ext_span,
53            syntax,
54            ..
55        } = args;
56
57        // Validate the `source` and `ext` value together, since they are
58        // related. In case `source` was used instead of `path`, the value
59        // of `ext` is merged into a synthetic `path` value here.
60        let path: Arc<Path> = match (&source, &ext) {
61            #[cfg(feature = "external-sources")]
62            (Source::Path(path), _) => {
63                config.find_template(path, None, None, Some(source_span.config_span()))?
64            }
65            (&Source::Source(_), Some(ext)) => {
66                PathBuf::from(format!("{}.{}", ast.ident, ext)).into()
67            }
68            (&Source::Source(_), None) => {
69                return Err(CompileError::no_file_info(
70                    #[cfg(not(feature = "code-in-doc"))]
71                    "must include `ext` attribute when using `source` attribute",
72                    #[cfg(feature = "code-in-doc")]
73                    "must include `ext` attribute when using `source` or `in_doc` attribute",
74                    None,
75                ));
76            }
77        };
78
79        // Validate syntax
80        let syntax = syntax.as_deref().map_or_else(
81            || Ok(config.syntaxes.get(config.default_syntax).unwrap()),
82            |s| {
83                config.syntaxes.get(s).ok_or_else(|| {
84                    CompileError::no_file_info(format_args!("syntax `{s}` is undefined"), None)
85                })
86            },
87        )?;
88
89        // Match extension against defined output formats
90
91        let escaping = escaping
92            .as_deref()
93            .or_else(|| path.extension().and_then(|s| s.to_str()))
94            .unwrap_or_default();
95
96        let escaper = config
97            .escapers
98            .iter()
99            .find_map(|(extensions, path)| {
100                extensions
101                    .contains(&Cow::Borrowed(escaping))
102                    .then_some(path.as_ref())
103            })
104            .ok_or_else(|| {
105                CompileError::no_file_info(
106                    format_args!(
107                        "no escaper defined for extension '{escaping}'. You can define an escaper \
108                        in the config file (named `askama.toml` by default). {}",
109                        MsgValidEscapers(&config.escapers),
110                    ),
111                    *ext_span,
112                )
113            })?;
114
115        let empty_punctuated = Punctuated::new();
116        let fields = match ast.data {
117            syn::Data::Struct(ref struct_) => {
118                if let syn::Fields::Named(fields) = &struct_.fields {
119                    &fields.named
120                } else {
121                    &empty_punctuated
122                }
123            }
124            syn::Data::Union(ref union_) => &union_.fields.named,
125            syn::Data::Enum(_) => &empty_punctuated,
126        }
127        .iter()
128        .map(|f| match &f.ident {
129            Some(ident) => ident.to_string(),
130            None => unreachable!("we checked that we are using a struct"),
131        })
132        .collect::<Vec<_>>();
133
134        Ok(TemplateInput {
135            template_span: *template_span,
136            ast,
137            enum_ast,
138            config,
139            syntax,
140            source,
141            source_span: source_span.clone(),
142            block: block.as_ref().map(|(block, span)| (block.as_str(), *span)),
143            blocks: blocks.as_slice(),
144            print: *print,
145            escaper,
146            path,
147            fields: fields.into(),
148        })
149    }

151    pub(crate) fn find_used_templates(
152        &self,
153        map: &mut HashMap<Arc<Path>, Arc<Parsed>>,
154    ) -> Result<(), CompileError> {
155        let (source, source_path) = match &self.source {
156            Source::Source(s) => (s.clone(), None),
157            #[cfg(feature = "external-sources")]
158            Source::Path(p) => (
159                get_template_source(&self.path, p.as_ref(), None)?,
160                Some(Arc::clone(&self.path)),
161            ),
162        };
163
164        #[cfg(feature = "external-sources")]
165        let mut dependency_graph = Vec::new();
166
167        #[cfg(feature = "external-sources")]
168        let mut original_paths: HashMap<Arc<Path>, String> = HashMap::default();
169        #[cfg(feature = "external-sources")]
170        if let Source::Path(p) = &self.source {
171            original_paths.insert(Arc::clone(&self.path), p.to_string());
172        }
173
174        let mut check = vec![(Arc::clone(&self.path), source, source_path)];
175        while let Some((path, source, source_path)) = check.pop() {
176            let parsed = match self.syntax.parse(Arc::clone(&source), source_path) {
177                Ok(parsed) => parsed,
178                Err(err) => {
179                    let msg = err
180                        .message
181                        .unwrap_or_else(|| "failed to parse template source".into());
182                    let file_path = err
183                        .file_path
184                        .as_deref()
185                        .unwrap_or(Path::new("<source attribute>"));
186                    let file_info =
187                        FileInfo::new(file_path, Some(&source), Some(&source[err.offset..]));
188                    return Err(CompileError::new(msg, Some(file_info)));
189                }
190            };
191
192            let mut top = true;
193            let mut nested = vec![parsed.nodes()];
194            while let Some(nodes) = nested.pop() {
195                for n in nodes {
196                    #[cfg(feature = "external-sources")]
197                    let mut add_to_check = |new_path: Arc<Path>,
198                                            original_path: &str|
199                     -> Result<(), CompileError> {
200                        if let std::collections::hash_map::Entry::Vacant(e) = map.entry(new_path) {
201                            // Add a dummy entry to `map` in order to prevent adding `path`
202                            // multiple times to `check`.
203                            let new_path = e.key();
204                            let source = parsed.source();
205                            let source = get_template_source(
206                                new_path,
207                                original_path,
208                                Some((
209                                    &path,
210                                    source,
211                                    n.span().as_suffix_of(source).unwrap_or_default(),
212                                )),
213                            )?;
214                            check.push((new_path.clone(), source, Some(new_path.clone())));
215                        }
216                        Ok(())
217                    };
218
219                    match &**n {
220                        Node::Extends(extends) if top => {
221                            #[cfg(not(feature = "external-sources"))]
222                            {
223                                return node_needs_external_sources(
224                                    "extends",
225                                    extends.span(),
226                                    &path,
227                                    &parsed,
228                                );
229                            }
230                            #[cfg(feature = "external-sources")]
231                            {
232                                let original_path = extends.path;
233                                let extends = self.config.find_template(
234                                    original_path,
235                                    Some(&path),
236                                    Some(FileInfo::of(extends.span(), &path, &parsed)),
237                                    Some(self.source_span.config_span()),
238                                )?;
239                                original_paths
240                                    .insert(Arc::clone(&extends), original_path.to_owned());
241                                let dependency_path = (path.clone(), extends.clone());
242                                if path == extends {
243                                    // We add the path into the graph to have a better looking error.
244                                    dependency_graph.push(dependency_path);
245                                    return cyclic_graph_error(&dependency_graph, &original_paths);
246                                } else if dependency_graph.contains(&dependency_path) {
247                                    return cyclic_graph_error(&dependency_graph, &original_paths);
248                                }
249                                dependency_graph.push(dependency_path);
250                                add_to_check(extends, original_path)?;
251                            }
252                        }
253                        Node::Macro(m) if top => {
254                            nested.push(&m.nodes);
255                        }
256                        Node::Import(import) if top => {
257                            #[cfg(not(feature = "external-sources"))]
258                            {
259                                return node_needs_external_sources(
260                                    "import",
261                                    import.span(),
262                                    &path,
263                                    &parsed,
264                                );
265                            }
266                            #[cfg(feature = "external-sources")]
267                            {
268                                let original_path = import.path;
269                                let import = self.config.find_template(
270                                    original_path,
271                                    Some(&path),
272                                    Some(FileInfo::of(import.span(), &path, &parsed)),
273                                    Some(self.source_span.config_span()),
274                                )?;
275                                original_paths
276                                    .insert(Arc::clone(&import), original_path.to_owned());
277                                add_to_check(import, original_path)?;
278                            }
279                        }
280                        Node::FilterBlock(f) => {
281                            nested.push(&f.nodes);
282                        }
283                        Node::Include(include) => {
284                            #[cfg(not(feature = "external-sources"))]
285                            {
286                                return node_needs_external_sources(
287                                    "include",
288                                    include.span(),
289                                    &path,
290                                    &parsed,
291                                );
292                            }
293                            #[cfg(feature = "external-sources")]
294                            {
295                                let original_path = include.path;
296                                let include = self.config.find_template(
297                                    original_path,
298                                    Some(&path),
299                                    Some(FileInfo::of(include.span(), &path, &parsed)),
300                                    Some(self.source_span.config_span()),
301                                )?;
302                                original_paths
303                                    .insert(Arc::clone(&include), original_path.to_owned());
304                                add_to_check(include, original_path)?;
305                            }
306                        }
307                        Node::BlockDef(b) => {
308                            nested.push(&b.nodes);
309                        }
310                        Node::If(i) => {
311                            for cond in &i.branches {
312                                nested.push(&cond.nodes);
313                            }
314                        }
315                        Node::Loop(l) => {
316                            nested.push(&l.body);
317                            nested.push(&l.else_nodes);
318                        }
319                        Node::Match(m) => {
320                            for arm in &m.arms {
321                                nested.push(&arm.nodes);
322                            }
323                        }
324                        Node::Call(c) => {
325                            nested.push(&c.nodes);
326                        }
327                        Node::Lit(_)
328                        | Node::Comment(_)
329                        | Node::Expr(_, _)
330                        | Node::Extends(_)
331                        | Node::Let(_)
332                        | Node::Compound(_)
333                        | Node::Declare(_)
334                        | Node::Import(_)
335                        | Node::Macro(_)
336                        | Node::Raw(_)
337                        | Node::Continue(_)
338                        | Node::Break(_) => {}
339                    }
340                }
341                top = false;
342            }
343            map.insert(path, parsed);
344        }
345        Ok(())
346    }
347}

362pub(crate) enum AnyTemplateArgs {
363    Struct(TemplateArgs),
364    Enum {
365        enum_args: Option<PartialTemplateArgs>,
366        vars_args: Vec<Option<PartialTemplateArgs>>,
367        has_default_impl: bool,
368    },
369}
370
371impl AnyTemplateArgs {
372    pub(crate) fn new(ast: &syn::DeriveInput) -> Result<Self, CompileError> {
373        let syn::Data::Enum(enum_data) = &ast.data else {
374            return Ok(Self::Struct(TemplateArgs::new(ast)?));
375        };
376
377        let enum_args = PartialTemplateArgs::new(ast, &ast.attrs, false)?;
378        let vars_args = enum_data
379            .variants
380            .iter()
381            .map(|variant| PartialTemplateArgs::new(ast, &variant.attrs, true))
382            .collect::<Result<Vec<_>, _>>()?;
383        if vars_args.is_empty() {
384            return Ok(Self::Struct(TemplateArgs::from_partial(ast, enum_args)?));
385        }
386
387        let mut needs_default_impl = vars_args.len();
388        let enum_source = enum_args.as_ref().and_then(|v| v.source.as_ref());
389        for (variant, var_args) in enum_data.variants.iter().zip(&vars_args) {
390            if var_args
391                .as_ref()
392                .and_then(|v| v.source.as_ref())
393                .or(enum_source)
394                .is_none()
395            {
396                return Err(CompileError::new_with_span(
397                    #[cfg(not(feature = "code-in-doc"))]
398                    "either all `enum` variants need a `path` or `source` argument, \
399                    or the `enum` itself needs a default implementation",
400                    #[cfg(feature = "code-in-doc")]
401                    "either all `enum` variants need a `path`, `source` or `in_doc` argument, \
402                    or the `enum` itself needs a default implementation",
403                    None,
404                    Some(variant.ident.span()),
405                ));
406            } else if !var_args.is_none() {
407                needs_default_impl -= 1;
408            }
409        }
410
411        Ok(Self::Enum {
412            enum_args,
413            vars_args,
414            has_default_impl: needs_default_impl > 0,
415        })
416    }
417
418    pub(crate) fn take_crate_name(&mut self) -> Option<ExprPath> {
419        match self {
420            AnyTemplateArgs::Struct(template_args) => template_args.crate_name.take(),
421            AnyTemplateArgs::Enum { enum_args, .. } => {
422                if let Some(PartialTemplateArgs { crate_name, .. }) = enum_args {
423                    crate_name.take()
424                } else {
425                    None
426                }
427            }
428        }
429    }

457    pub(crate) fn from_partial(
458        ast: &syn::DeriveInput,
459        args: Option<PartialTemplateArgs>,
460    ) -> Result<Self, CompileError> {
461        let Some(args) = args else {
462            return Err(CompileError::new_with_span(
463                "no attribute `template` found",
464                None,
465                Some(ast.ident.span()),
466            ));
467        };
468        Ok(Self {
469            template_span: args.template_span,
470            source: match args.source {
471                #[cfg(feature = "external-sources")]
472                Some(PartialTemplateArgsSource::Path(s)) => {
473                    (Source::Path(s.value().into()), SourceSpan::from_path(s)?)
474                }
475                Some(PartialTemplateArgsSource::Source(s)) => {
476                    let (source, span) = SourceSpan::from_source(s)?;
477                    (Source::Source(source.into()), span)
478                }
479                #[cfg(feature = "code-in-doc")]
480                Some(PartialTemplateArgsSource::InDoc(span, source)) => {
481                    (source, SourceSpan::CodeInDoc(span))
482                }
483                None => {
484                    return Err(CompileError::no_file_info(
485                        #[cfg(not(feature = "code-in-doc"))]
486                        "specify one template argument `path` or `source`",
487                        #[cfg(feature = "code-in-doc")]
488                        "specify one template argument `path`, `source` or `in_doc`",
489                        Some(args.template_span),
490                    ));
491                }
492            },
493            block: args.block.map(|value| (value.value(), value.span())),
494            blocks: args
495                .blocks
496                .unwrap_or_default()
497                .into_iter()
498                .map(|value| Block {
499                    name: value.value(),
500                })
501                .collect(),
502            print: args.print.unwrap_or_default(),
503            escaping: args.escape.map(|value| value.value()),
504            ext: args.ext.as_ref().map(|value| value.value()),
505            ext_span: args.ext.as_ref().map(|value| value.span()),
506            syntax: args.syntax.map(|value| value.value()),
507            config: args.config.as_ref().map(|value| value.value()),
508            crate_name: args.crate_name,
509            whitespace: args.whitespace,
510            config_span: args.config.as_ref().map(|value| value.span()),
511        })
512    }

537/// Try to find the source in the comment, in a `askama` code block.
538///
539/// This is only done if no path or source was given in the `#[template]` attribute.
540#[cfg(feature = "code-in-doc")]
541fn source_from_docs(
542    span: Span,
543    docs: &[&Attribute],
544    ast: &syn::DeriveInput,
545) -> Result<(Source, Option<Span>), CompileError> {
546    let (source_span, source) = collect_comment_blocks(span, docs, ast)?;
547    let source = strip_common_ws_prefix(source);
548    let source = collect_askama_code_blocks(span, ast, source)?;
549    Ok((source, source_span))
550}

552#[cfg(feature = "code-in-doc")]
553fn collect_comment_blocks(
554    span: Span,
555    docs: &[&Attribute],
556    ast: &syn::DeriveInput,
557) -> Result<(Option<Span>, String), CompileError> {
558    let mut source_span: Option<Span> = None;
559    let mut assign_span = |kv: &syn::MetaNameValue| {
560        // FIXME: uncomment once <https://github.com/rust-lang/rust/issues/54725> is stable
561        // let new_span = kv.path.span();
562        // source_span = Some(match source_span {
563        //     Some(cur_span) => cur_span.join(new_span).unwrap_or(cur_span),
564        //     None => new_span,
565        // });
566
567        if source_span.is_none() {
568            source_span = Some(kv.path.span());
569        }
570    };
571
572    let mut source = String::new();
573    for a in docs {
574        // is a comment?
575        let Meta::NameValue(kv) = &a.meta else {
576            continue;
577        };
578        if !kv.path.is_ident("doc") {
579            continue;
580        }
581
582        // is an understood comment, e.g. not `#[doc = inline_str(…)]`
583        let mut value = &kv.value;
584        let value = loop {
585            match value {
586                Expr::Lit(lit) => break lit,
587                Expr::Group(group) => value = &group.expr,
588                _ => continue,
589            }
590        };
591        let Lit::Str(value) = &value.lit else {
592            continue;
593        };
594
595        assign_span(kv);
596        source.push_str(value.value().as_str());
597        source.push('\n');
598    }
599    if source.is_empty() {
600        return Err(no_askama_code_block(span, ast));
601    }
602
603    Ok((source_span, source))
604}

606#[cfg(feature = "code-in-doc")]
607fn no_askama_code_block(span: Span, ast: &syn::DeriveInput) -> CompileError {
608    let kind = match &ast.data {
609        syn::Data::Struct(_) => "struct",
610        syn::Data::Enum(_) => "enum",
611        // actually unreachable: `union`s are rejected by `TemplateArgs::new()`
612        syn::Data::Union(_) => "union",
613    };
614    CompileError::no_file_info(
615        format_args!(
616            "when using `in_doc` with the value `true`, the {kind}'s documentation needs a \
617             `askama` code block"
618        ),
619        Some(span),
620    )
621}

623#[cfg(feature = "code-in-doc")]
624fn strip_common_ws_prefix(source: String) -> String {
625    let mut common_prefix_iter = source
626        .lines()
627        .filter_map(|s| Some(&s[..s.find(|c: char| !c.is_ascii_whitespace())?]));
628    let mut common_prefix = common_prefix_iter.next().unwrap_or_default();
629    for p in common_prefix_iter {
630        if common_prefix.is_empty() {
631            break;
632        }
633        let ((pos, _), _) = common_prefix
634            .char_indices()
635            .zip(p.char_indices())
636            .take_while(|(l, r)| l == r)
637            .last()
638            .unwrap_or_default();
639        common_prefix = &common_prefix[..pos];
640    }
641    if common_prefix.is_empty() {
642        return source;
643    }
644
645    source
646        .lines()
647        .flat_map(|s| [s.get(common_prefix.len()..).unwrap_or_default(), "\n"])
648        .collect()
649}

651#[cfg(feature = "code-in-doc")]
652fn collect_askama_code_blocks(
653    span: Span,
654    ast: &syn::DeriveInput,
655    source: String,
656) -> Result<Source, CompileError> {
657    use pulldown_cmark::{CodeBlockKind, Event, Parser, Tag, TagEnd};
658
659    let mut tmpl_source = String::new();
660    let mut in_askama_code = false;
661    let mut had_askama_code = false;
662    for e in Parser::new(&source) {
663        match (in_askama_code, e) {
664            (false, Event::Start(Tag::CodeBlock(CodeBlockKind::Fenced(s))))
665                if s.split(",")
666                    .any(|s| JINJA_EXTENSIONS.contains(&s.trim_ascii())) =>
667            {
668                in_askama_code = true;
669                had_askama_code = true;
670            }
671            (true, Event::End(TagEnd::CodeBlock)) => in_askama_code = false,
672            (true, Event::Text(text)) => tmpl_source.push_str(&text),
673            _ => {}
674        }
675    }
676    if !had_askama_code {
677        return Err(no_askama_code_block(span, ast));
678    }
679
680    if tmpl_source.ends_with('\n') {
681        tmpl_source.pop();
682    }
683    Ok(Source::Source(tmpl_source.into()))
684}

716#[cfg(feature = "external-sources")]
717fn cyclic_graph_error(
718    dependency_graph: &[(Arc<Path>, Arc<Path>)],
719    original_paths: &HashMap<Arc<Path>, String>,
720) -> Result<(), CompileError> {
721    let fmt_path = |p: &Arc<Path>| match original_paths.get(p) {
722        Some(orig) => format!("{orig:#?} (absolute path: {:#?})", p.display()),
723        None => format!("{:#?}", p.display()),
724    };
725    Err(CompileError::no_file_info(
726        format_args!(
727            "cyclic dependency in graph {:#?}",
728            dependency_graph
729                .iter()
730                .map(|e| format!("{} --> {}", fmt_path(&e.0), fmt_path(&e.1)))
731                .collect::<Vec<String>>()
732        ),
733        None,
734    ))
735}

737#[cfg(feature = "external-sources")]
738pub(crate) fn get_template_source(
739    tpl_path: &Arc<Path>,
740    original_path: &str,
741    import_from: Option<(&Arc<Path>, &str, &str)>,
742) -> Result<Arc<str>, CompileError> {
743    static CACHE: std::sync::OnceLock<crate::OnceMap<Arc<Path>, Arc<str>>> =
744        std::sync::OnceLock::new();
745
746    CACHE
747        .get_or_init(crate::OnceMap::default)
748        .get_or_try_insert(
749            tpl_path,
750            |tpl_path| match std::fs::read_to_string(tpl_path) {
751                Ok(mut source) => {
752                    if source.ends_with('\n') {
753                        let _ = source.pop();
754                    }
755                    Ok((Arc::clone(tpl_path), Arc::from(source)))
756                }
757                Err(err) => Err(CompileError::new(
758                    format_args!(
759                        "unable to open template file '{}' (absolute path: '{}'): {err}",
760                        original_path,
761                        tpl_path.to_str().unwrap(),
762                    ),
763                    import_from.map(|(node_file, file_source, node_source)| {
764                        FileInfo::new(node_file, Some(file_source), Some(node_source))
765                    }),
766                )),
767            },
768            Arc::clone,
769        )
770}
