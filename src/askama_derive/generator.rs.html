25pub(crate) fn template_to_string(
26    buf: &mut Buffer,
27    input: &TemplateInput<'_>,
28    contexts: &HashMap<&Arc<Path>, Context<'_>>,
29    heritage: Option<&Heritage<'_, '_>>,
30    tmpl_kind: TmplKind<'_>,
31) -> Result<SizeHint, CompileError> {
32    let generator = Generator::new(
33        input,
34        contexts,
35        heritage,
36        MapChain::default(),
37        input.block.is_some(),
38        BlockInfo::new(),
39    );
40    let size_hint = match generator.impl_template(buf, tmpl_kind) {
41        Err(mut err) if err.span.is_none() => {
42            err.span = Some(input.source_span.config_span());
43            Err(err)
44        }
45        result => result,
46    }?;
47
48    if tmpl_kind == TmplKind::Struct {
49        impl_everything(input.ast, buf);
50    }
51    Ok(size_hint)
52}

75struct BlockInfo {
76    block_name: &'static str,
77    level: usize,
78}
79
80impl BlockInfo {
81    fn new() -> Self {
82        Self {
83            block_name: "",
84            level: 0,
85        }
86    }
87
88    // FIXME: Instead of this error-prone API, we should use something relying on `Drop` to
89    // decrement, or use a callback which would decrement on exit.
90    fn increase(&mut self, block_name: &'static str) {
91        if self.level == 0 {
92            self.block_name = block_name;
93        }
94        self.level += 1;
95    }
96
97    fn decrease(&mut self) {
98        self.level -= 1;
99    }
100}

102struct Generator<'a, 'h> {
103    /// The template input state: original struct AST and attributes
104    input: &'a TemplateInput<'a>,
105    /// All contexts, keyed by the package-relative template path
106    contexts: &'a HashMap<&'a Arc<Path>, Context<'a>>,
107    /// The heritage contains references to blocks and their ancestry
108    heritage: Option<&'h Heritage<'a, 'h>>,
109    /// Variables accessible directly from the current scope (not redirected to context)
110    locals: MapChain<'a>,
111    /// Suffix whitespace from the previous literal. Will be flushed to the
112    /// output buffer unless suppressed by whitespace suppression on the next
113    /// non-literal.
114    next_ws: Option<WithSpan<&'a str>>,
115    /// Whitespace suppression from the previous non-literal. Will be used to
116    /// determine whether to flush prefix whitespace from the next literal.
117    skip_ws: Whitespace,
118    /// If currently in a block, this will contain the name of a potential parent block
119    super_block: Option<(&'a str, usize)>,
120    /// Buffer for writable
121    buf_writable: WritableBuffer<'a>,
122    /// Used in blocks to check if we are inside a filter/let block.
123    is_in_block: BlockInfo,
124    /// Set of called macros we are currently in. Used to prevent (indirect) recursions.
125    seen_callers: Vec<(&'a Macro<'a>, Option<FileInfo<'a>>)>,
126    /// The directory path of the calling file.
127    caller_dir: CallerDir,
128}

130enum CallerDir {
131    Valid(PathBuf),
132    Invalid,
133    Unresolved,

163    fn rel_path<'p>(&mut self, path: &'p Path) -> Cow<'p, Path> {
164        self.caller_dir()
165            .and_then(|caller_dir| {
166                diff_paths(path, caller_dir, std::env::var("CARGO_MANIFEST_DIR").ok())
167            })
168            .map_or(Cow::Borrowed(path), Cow::Owned)
169    }
170
171    fn caller_dir(&mut self) -> Option<&Path> {
172        match self.caller_dir {
173            CallerDir::Valid(ref caller_dir) => return Some(caller_dir.as_path()),
174            CallerDir::Invalid => return None,
175            CallerDir::Unresolved => {}
176        }
177
178        if proc_macro::is_available()
179            && let Some(mut local_file) = proc_macro::Span::call_site().local_file()
180        {
181            local_file.pop();
182            if !local_file.is_absolute() {
183                local_file = Path::new(".").join(local_file);
184            }
185
186            self.caller_dir = CallerDir::Valid(clean_path(&local_file));
187            match &self.caller_dir {
188                CallerDir::Valid(caller_dir) => Some(caller_dir.as_path()),
189                _ => None, // unreachable
190            }
191        } else {
192            self.caller_dir = CallerDir::Invalid;
193            None
194        }
195    }

197    // Implement `Template` for the given context struct.
198    fn impl_template(
199        mut self,
200        buf: &mut Buffer,
201        tmpl_kind: TmplKind<'a>,
202    ) -> Result<SizeHint, CompileError> {
203        let ctx = &self.contexts[&self.input.path];
204
205        let span = Span::call_site();
206        let target = match tmpl_kind {
207            TmplKind::Struct => quote_spanned!(span=> askama::Template),
208            TmplKind::Variant => quote_spanned!(span=> askama::helpers::EnumVariantTemplate),
209            TmplKind::Block(trait_name) => field_new(trait_name, span),
210        };
211
212        let mut paths_ts = TokenStream::new();
213
214        if let Some(full_config_path) = &self.input.config.full_config_path {
215            let full_config_path = self.rel_path(full_config_path).display().to_string();
216            paths_ts.extend(quote_spanned!(span =>
217                const _: &[askama::helpers::core::primitive::u8] =
218                    askama::helpers::core::include_bytes!(#full_config_path);
219            ));
220        }
221
222        // Make sure the compiler understands that the generated code depends on the template files.
223        let mut paths = self
224            .contexts
225            .iter()
226            .map(|(path, _ctx)| {
227                (
228                    &***path,
229                    #[cfg(not(feature = "external-sources"))]
230                    (),
231                    #[cfg(feature = "external-sources")]
232                    _ctx,
233                )
234            })
235            .filter(|&(path, _)| {
236                // Skip the fake path of templates defined in rust source.
237                match self.input.source {
238                    #[cfg(feature = "external-sources")]
239                    Source::Path(_) => true,
240                    Source::Source(_) => *path != *self.input.path,
241                }
242            })
243            .collect::<Vec<_>>();
244        paths.sort_by_key(|&(path, _)| path);
245        for (path, _ctx) in paths {
246            let path = self.rel_path(path).display().to_string();
247            paths_ts.extend(quote_spanned!(span=>
248                const _: &[askama::helpers::core::primitive::u8] =
249                    askama::helpers::core::include_bytes!(#path);
250            ));
251
252            #[cfg(all(feature = "external-sources", feature = "nightly-spans"))]
253            _ctx.resolve_path(&path);
254        }
255
256        let mut content = Buffer::new();
257        let size_hint = self.impl_template_inner(ctx, &mut content)?;
258        let content = content.into_token_stream();
259
260        let mut size_hint_s = TokenStream::new();
261        if tmpl_kind == TmplKind::Struct {
262            size_hint_s = quote_spanned!(span=>
263                const SIZE_HINT: askama::helpers::core::primitive::usize = #size_hint;
264            );
265        }
266
267        write_header(self.input.ast, buf, target);
268        let var_writer = crate::var_writer();
269        let var_values = crate::var_values();
270        quote_into!(buf, span, { {
271            fn render_into_with_values(
272                &self,
273                #var_writer: &mut dyn askama::helpers::core::fmt::Write,
274                #var_values: &dyn askama::Values,
275            ) -> askama::Result<()> {
276                #[allow(unused_imports)]
277                use askama::{
278                    filters::{AutoEscape as _, WriteWritable as _},
279                    helpers::{ResultConverter as _, core::fmt::Write as _},
280                };
281
282                #paths_ts
283                #content
284                askama::Result::Ok(())
285            }
286            #size_hint_s
287        } });
288
289        for block in self.input.blocks {
290            self.impl_block(buf, block)?;
291        }
292
293        Ok(size_hint)
294    }

296    fn impl_block(
297        &self,
298        buf: &mut Buffer,
299        block: &crate::input::Block,
300    ) -> Result<(), CompileError> {
301        // RATIONALE: `*self` must be the input type, implementation details should not leak:
302        // - impl Self { fn as_block(self) } ->
303        // - struct __Askama__Self__as__block__Wrapper { this: self } ->
304        // - impl Template for __Askama__Self__as__block__Wrapper { fn render_into_with_values() } ->
305        // - impl __Askama__Self__as__block for Self { render_into_with_values() }
306
307        use syn::{GenericParam, Ident, Lifetime, LifetimeParam, Token};
308
309        let span = Span::call_site();
310        let ident = &self.input.ast.ident;
311
312        let doc = format!(
313            "A sub-template that renders only the block `{}` of [`{ident}`].",
314            block.name
315        );
316        let method_name = format!("as_{}", block.name);
317        let trait_name = format!("__Askama__{ident}__as__{}", block.name);
318        let wrapper_name = format!("__Askama__{ident}__as__{}__Wrapper", block.name);
319        let self_lt_name = format!("'__Askama__{ident}__as__{}__self", block.name);
320
321        let method_id = Ident::new(&method_name, span);
322        let trait_id = Ident::new(&trait_name, span);
323        let wrapper_id = Ident::new(&wrapper_name, span);
324        let self_lt = Lifetime::new(&self_lt_name, span);
325
326        // generics of the input with an additional lifetime to capture `self`
327        let mut wrapper_generics = self.input.ast.generics.clone();
328        if wrapper_generics.lt_token.is_none() {
329            wrapper_generics.lt_token = Some(Token![<](span));
330            wrapper_generics.gt_token = Some(Token![>](span));
331        }
332        wrapper_generics.params.insert(
333            0,
334            GenericParam::Lifetime(LifetimeParam::new(self_lt.clone())),
335        );
336
337        let (impl_generics, ty_generics, where_clause) = self.input.ast.generics.split_for_impl();
338        let (wrapper_impl_generics, wrapper_ty_generics, wrapper_where_clause) =
339            wrapper_generics.split_for_impl();
340
341        let input = TemplateInput {
342            block: Some((&block.name, span)),
343            blocks: &[],
344            ..self.input.clone()
345        };
346        let mut template_buf = Buffer::new();
347        let size_hint = template_to_string(
348            &mut template_buf,
349            &input,
350            self.contexts,
351            self.heritage,
352            TmplKind::Block(&trait_name),
353        )?;
354
355        let template_buf = template_buf.into_token_stream();
356        quote_into!(buf, span, {
357            #[allow(missing_docs, non_camel_case_types, non_snake_case, unreachable_pub)]
358            const _: () = {
359                #template_buf
360
361                pub trait #trait_id {
362                    fn render_into_with_values(
363                        &self,
364                        writer: &mut dyn askama::helpers::core::fmt::Write,
365                        values: &dyn askama::Values,
366                    ) -> askama::Result<()>;
367                }
368
369                impl #impl_generics #ident #ty_generics #where_clause {
370                    #[inline]
371                    #[doc = #doc]
372                    pub fn #method_id(&self) -> impl askama::Template + '_ {
373                        #wrapper_id {
374                            this: self,
375                        }
376                    }
377                }
378
379                #[askama::helpers::core::prelude::rust_2021::derive(
380                    askama::helpers::core::prelude::rust_2021::Clone,
381                    askama::helpers::core::prelude::rust_2021::Copy
382                )]
383                pub struct #wrapper_id #wrapper_generics #wrapper_where_clause {
384                    this: &#self_lt #ident #ty_generics,
385                }
386
387                impl #wrapper_impl_generics askama::Template
388                for #wrapper_id #wrapper_ty_generics #wrapper_where_clause {
389                    #[inline]
390                    fn render_into_with_values(
391                        &self,
392                        writer: &mut dyn askama::helpers::core::fmt::Write,
393                        values: &dyn askama::Values
394                    ) -> askama::Result<()> {
395                        <_ as #trait_id>::render_into_with_values(self.this, writer, values)
396                    }
397
398                    const SIZE_HINT: askama::helpers::core::primitive::usize = #size_hint;
399                }
400
401                // cannot use `crate::integrations::impl_fast_writable()` w/o cloning the struct
402                impl #wrapper_impl_generics askama::FastWritable
403                for #wrapper_id #wrapper_ty_generics #wrapper_where_clause {
404                    #[inline]
405                    fn write_into(
406                        &self,
407                        dest: &mut dyn askama::helpers::core::fmt::Write,
408                        values: &dyn askama::Values,
409                    ) -> askama::Result<()> {
410                        <_ as askama::Template>::render_into_with_values(self, dest, values)
411                    }
412                }
413
414                // cannot use `crate::integrations::impl_display()` w/o cloning the struct
415                impl #wrapper_impl_generics askama::helpers::core::fmt::Display
416                for #wrapper_id #wrapper_ty_generics #wrapper_where_clause {
417                    #[inline]
418                    fn fmt(
419                        &self,
420                        f: &mut askama::helpers::core::fmt::Formatter<'_>
421                    ) -> askama::helpers::core::fmt::Result {
422                        <_ as askama::Template>::render_into(self, f)
423                            .map_err(|_| askama::helpers::core::fmt::Error)
424                    }
425                }
426            };
427        });
428
429        Ok(())
430    }

432    fn is_var_defined(&self, var_name: &str) -> bool {
433        self.locals.get_any(var_name).is_some() || self.input.fields.iter().any(|f| f == var_name)
434    }
435
436    /// Like [`is_var_defined()`], but not true for a forward declaration `{% let var %}`.
437    fn is_var_assigned(&self, var_name: &str) -> bool {
438        if let Some(meta) = self.locals.get(var_name) {
439            meta.initialized
440        } else {
441            self.input.fields.iter().any(|f| f == var_name)
442        }
443    }

469/// In here, we inspect in the expression if it is a literal, and if it is, whether it
470/// can be escaped at compile time.
471fn compile_time_escape<'a>(expr: &WithSpan<Box<Expr<'a>>>, escaper: &str) -> Option<Writable<'a>> {
472    // we only optimize for known escapers
473    enum OutputKind {
474        Html,
475        Text,
476    }
477
478    // we only optimize for known escapers
479    let output = match escaper.strip_prefix("askama::filters::")? {
480        "Html" => OutputKind::Html,
481        "Text" => OutputKind::Text,
482        _ => return None,
483    };
484
485    // for now, we only escape strings, chars, numbers, and bools at compile time
486    let value = match ***expr {
487        Expr::StrLit(StrLit {
488            prefix: None,
489            content,
490            ..
491        }) => {
492            if content.find('\\').is_none() {
493                // if the literal does not contain any backslashes, then it does not need unescaping
494                Cow::Borrowed(content)
495            } else {
496                // the input could be string escaped if it contains any backslashes
497                let input = format!(r#""{content}""#);
498                let input = input.parse().ok()?;
499                let input = syn::parse2::<syn::LitStr>(input).ok()?;
500                Cow::Owned(input.value())
501            }
502        }
503        Expr::CharLit(CharLit {
504            prefix: None,
505            content,
506        }) => {
507            if content.find('\\').is_none() {
508                // if the literal does not contain any backslashes, then it does not need unescaping
509                Cow::Borrowed(content)
510            } else {
511                // the input could be string escaped if it contains any backslashes
512                let input = format!(r#"'{content}'"#);
513                let input = input.parse().ok()?;
514                let input = syn::parse2::<syn::LitChar>(input).ok()?;
515                Cow::Owned(input.value().to_string())
516            }
517        }
518        Expr::NumLit(_, value) => {
519            enum NumKind {
520                Int(Option<IntKind>),
521                Float(Option<FloatKind>),
522            }
523
524            let (orig_value, kind) = match value {
525                Num::Int(value, kind) => (value, NumKind::Int(kind)),
526                Num::Float(value, kind) => (value, NumKind::Float(kind)),
527            };
528            let value = match orig_value.chars().any(|c| c == '_') {
529                true => Cow::Owned(orig_value.chars().filter(|&c| c != '_').collect()),
530                false => Cow::Borrowed(orig_value),
531            };
532
533            fn int<T: ToString, E>(
534                from_str_radix: impl Fn(&str, u32) -> Result<T, E>,
535                value: &str,
536            ) -> Option<String> {
537                let mut chars = value.chars();
538                let (value, radix) = if let Some('0') = chars.next() {
539                    match chars.next() {
540                        Some('x') => (&value[2..], 16),
541                        Some('o') => (&value[2..], 8),
542                        Some('b') => (&value[2..], 2),
543                        Some(_) | None => (value, 10),
544                    }
545                } else {
546                    (value, 10)
547                };
548
549                Some(from_str_radix(value, radix).ok()?.to_string())
550            }
551
552            let value = match kind {
553                NumKind::Int(Some(IntKind::I8)) => int(i8::from_str_radix, &value)?,
554                NumKind::Int(Some(IntKind::I16)) => int(i16::from_str_radix, &value)?,
555                NumKind::Int(Some(IntKind::I32)) => int(i32::from_str_radix, &value)?,
556                NumKind::Int(Some(IntKind::I64)) => int(i64::from_str_radix, &value)?,
557                NumKind::Int(Some(IntKind::I128)) => int(i128::from_str_radix, &value)?,
558                NumKind::Int(Some(IntKind::Isize)) => int(TargetIsize::from_str_radix, &value)?,
559                NumKind::Int(Some(IntKind::U8)) => int(u8::from_str_radix, &value)?,
560                NumKind::Int(Some(IntKind::U16)) => int(u16::from_str_radix, &value)?,
561                NumKind::Int(Some(IntKind::U32)) => int(u32::from_str_radix, &value)?,
562                NumKind::Int(Some(IntKind::U64)) => int(u64::from_str_radix, &value)?,
563                NumKind::Int(Some(IntKind::U128)) => int(u128::from_str_radix, &value)?,
564                NumKind::Int(Some(IntKind::Usize)) => int(TargetUsize::from_str_radix, &value)?,
565                NumKind::Int(None) => match value.starts_with('-') {
566                    true => int(i128::from_str_radix, &value)?,
567                    false => int(u128::from_str_radix, &value)?,
568                },
569                NumKind::Float(Some(FloatKind::F32)) => value.parse::<f32>().ok()?.to_string(),
570                NumKind::Float(Some(FloatKind::F64) | None) => {
571                    value.parse::<f64>().ok()?.to_string()
572                }
573                // FIXME: implement once `f16` and `f128` are available
574                NumKind::Float(Some(FloatKind::F16 | FloatKind::F128)) => return None,
575            };
576            match value == orig_value {
577                true => Cow::Borrowed(orig_value),
578                false => Cow::Owned(value),
579            }
580        }
581        Expr::BoolLit(true) => Cow::Borrowed("true"),
582        Expr::BoolLit(false) => Cow::Borrowed("false"),
583        _ => return None,
584    };
585
586    // escape the un-string-escaped input using the selected escaper
587    Some(Writable::Lit(match output {
588        OutputKind::Text => WithSpan::new(value, expr.span()),
589        OutputKind::Html => {
590            let mut escaped = String::with_capacity(value.len() + 20);
591            write_escaped_str(&mut escaped, &value).ok()?;
592            match escaped == value {
593                true => WithSpan::new(value, expr.span()),
594                false => WithSpan::new(Cow::Owned(escaped), expr.span()),
595            }
596        }
597    }))
598}

631    Negative,
632}
633
634impl<'a> LocalMeta<'a> {
635    /// Variable declaration only - no value yet.
636    const fn var_decl() -> Self {
637        Self::Variable(LocalVariableMeta {
638            refs: None,
639            initialized: false,
640        })
641    }
642
643    /// Variable definition - fully initialized.
644    const fn var_def() -> Self {
645        Self::Variable(LocalVariableMeta {
646            refs: None,
647            initialized: true,
648        })
649    }
650
651    /// Variable referencing another
652    const fn var_with_ref(refs: String) -> Self {
653        Self::Variable(LocalVariableMeta {
654            refs: Some(refs),
655            initialized: true,
656        })
657    }
658
659    /// Special variable aliasing a `caller()`
660    const fn caller(def: &'a Call<'a>, call_ctx: Context<'a>) -> Self {
661        Self::CallerAlias(LocalCallerMeta { def, call_ctx })
662    }
663}

665#[derive(Debug)]
666struct MapChain<'a> {
667    scopes: Vec<HashMap<Cow<'a, str>, LocalMeta<'a>>>,
668}
669
670impl<'a> MapChain<'a> {
671    fn new_empty() -> Self {
672        Self { scopes: vec![] }
673    }
674
675    /// Iterates the scopes in reverse and searches for a local variable (of any kind) with the given key.
676    ///
677    /// # Returns
678    /// - `Some(LocalMeta)` if any kind of local entry (except negative) with the key was found.
679    /// - `None` otherwise
680    fn get_any<'b>(&'b self, key: &str) -> Option<&'b LocalMeta<'a>> {
681        match self.scopes.iter().rev().find_map(|set| set.get(key)) {
682            Some(LocalMeta::Negative) => None,
683            Some(local) => Some(local),
684            _ => None,
685        }
686    }
687
688    /// Iterates the scopes in reverse and searches for a local variable with the given key.
689    ///
690    /// # Returns
691    /// - `Some(LocalVariableMeta)` if the first encountered entry for key was a variable
692    /// - `None` otherwise
693    fn get<'b>(&'b self, key: &str) -> Option<&'b LocalVariableMeta> {
694        match self.scopes.iter().rev().find_map(|set| set.get(key)) {
695            Some(LocalMeta::Variable(var)) => Some(var),
696            _ => None,
697        }
698    }
699
700    /// Iterates the scopes in reverse and searches for a `CallerAlias`
701    ///
702    /// # Returns
703    /// - `Some(LocalCallerMeta)` if the first encountered entry for key was a caller alias
704    /// - `None` otherwise
705    fn get_caller<'b>(&'b self, key: &str) -> Option<&'b LocalCallerMeta<'a>> {
706        match self.scopes.iter().rev().find_map(|set| set.get(key)) {
707            Some(LocalMeta::CallerAlias(caller)) => Some(caller),
708            _ => None,
709        }
710    }
711
712    fn is_current_empty(&self) -> bool {
713        self.scopes.last().unwrap().is_empty()
714    }
715
716    fn insert(&mut self, key: Cow<'a, str>, val: LocalMeta<'a>) {
717        self.scopes.last_mut().unwrap().insert(key, val);
718
719        // Note that if `insert` returns `Some` then it implies
720        // an identifier is reused. For e.g. `{% macro f(a, a) %}`
721        // and `{% let (a, a) = ... %}` then this results in a
722        // generated template, which when compiled fails with the
723        // compile error "identifier `a` used more than once".
724    }
725
726    fn insert_with_default(&mut self, key: Cow<'a, str>) {
727        self.insert(key, LocalMeta::var_decl());
728    }
729
730    fn resolve(&self, name: &str) -> Option<String> {
731        self.get(&Cow::Borrowed(name)).map(|meta| match &meta.refs {
732            Some(expr) => expr.clone(),
733            None => name.to_string(),
734        })
735    }
736
737    fn resolve_or_self(&self, name: &str) -> String {
738        self.resolve(name).unwrap_or_else(|| format!("self.{name}"))
739    }
740
741    fn stack_push(&mut self) {
742        self.scopes.push(HashMap::default());
743    }
744
745    fn stack_pop(&mut self) {
746        self.scopes.pop().unwrap();
747    }
748}
749
750impl Default for MapChain<'_> {
751    fn default() -> Self {
752        Self {
753            scopes: vec![HashMap::default()],
754        }
755    }
756}

758/// Returns `true` if enough assumptions can be made,
759/// to determine that `self` is copyable.
760fn is_copyable(expr: &Expr<'_>) -> bool {
761    is_copyable_within_op(expr, false)
762}
763
764fn is_copyable_within_op(expr: &Expr<'_>, within_op: bool) -> bool {
765    match expr {
766        Expr::BoolLit(_)
767        | Expr::NumLit(_, _)
768        | Expr::StrLit(_)
769        | Expr::CharLit(_)
770        | Expr::BinOp(_)
771        | Expr::Range(..) => true,
772        Expr::Unary(.., expr) => is_copyable_within_op(expr, true),
773        Expr::NamedArgument(_, expr) => is_copyable_within_op(expr, true),
774        // The result of a call likely doesn't need to be borrowed,
775        // as in that case the call is more likely to return a
776        // reference in the first place then.
777        Expr::Call { .. } | Expr::Path(..) | Expr::Filter(..) | Expr::RustMacro(..) => true,
778        // If the `expr` is within a `Unary` or `BinOp` then
779        // an assumption can be made that the operand is copy.
780        // If not, then the value is moved and adding `.clone()`
781        // will solve that issue. However, if the operand is
782        // implicitly borrowed, then it's likely not even possible
783        // to get the template to compile.
784        _ => within_op && is_associated_item_self(expr),
785    }
786}
787
788/// Returns `true` if this is an `AssociatedItem` where the `obj` is `"self"`.
789fn is_associated_item_self(mut expr: &Expr<'_>) -> bool {
790    loop {
791        match expr {
792            Expr::AssociatedItem(obj, _) if matches!(***obj, Expr::Var("self")) => return true,
793            Expr::AssociatedItem(obj, _) if matches!(***obj, Expr::AssociatedItem(..)) => {
794                expr = obj
795            }
796            _ => return false,
797        }
798    }
799}

836macro_rules! make_token_match {
837    ($op:ident @ $span:ident => $($tt:tt)+) => {
838        match $op {
839            $(stringify!($tt) => Token![$tt]($span).into_token_stream(),)+
840            _ => unreachable!(),
841        }
842    };
843}
844
845#[inline]
846#[track_caller]
847fn logic_op(op: &str, span: proc_macro2::Span) -> TokenStream {
848    make_token_match!(op @ span => && || ^)
849}
850
851#[inline]
852#[track_caller]
853fn unary_op(op: &str, span: proc_macro2::Span) -> TokenStream {
854    make_token_match!(op @ span => - ! * &)
855}
856
857#[inline]
858#[track_caller]
859fn range_op(op: &str, span: proc_macro2::Span) -> TokenStream {
860    make_token_match!(op @ span => .. ..=)
861}
862
863#[inline]
864#[track_caller]
865fn binary_op(op: &str, span: proc_macro2::Span) -> TokenStream {
866    make_token_match!(
867        op @ span =>
868        * / % + - << >> & ^ | == != < > <= >= && || .. ..=
869        = += -= *= /= %= &= |= ^= <<= >>=
870    )
