227        const SYNTAX_BEHAVIOR_ALLOW_UNMATCHED_CLOSE_SUBEXP

231            = onig_sys::ONIG_SYN_ALLOW_INVALID_INTERVAL;
232        /// `{,n} => {0,n}`
233        const SYNTAX_BEHAVIOR_ALLOW_INTERVAL_LOW_ABBREV

238        /// `(?<=a|bc)`

241        /// See Oniguruma documenation
242        const SYNTAX_BEHAVIOR_CAPTURE_ONLY_NAMED_GROUP
243            = onig_sys::ONIG_SYN_CAPTURE_ONLY_NAMED_GROUP;
244        /// `(?<x>)(?<x>)`
245        const SYNTAX_BEHAVIOR_ALLOW_MULTIPLEX_DEFINITION_NAME
246            = onig_sys::ONIG_SYN_ALLOW_MULTIPLEX_DEFINITION_NAME;
247        /// `a{n}?=(?:a{n})?`
248        const SYNTAX_BEHAVIOR_FIXED_INTERVAL_IS_GREEDY_ONLY

250        /// `[^...]`
251        const SYNTAX_BEHAVIOR_NOT_NEWLINE_IN_NEGATIVE_CC
252            = onig_sys::ONIG_SYN_NOT_NEWLINE_IN_NEGATIVE_CC;
253        /// `[..\w..] etc..`
254        const SYNTAX_BEHAVIOR_BACKSLASH_ESCAPE_IN_CC

256        /// `[0-9-a]=[0-9\-a]`
257        const SYNTAX_BEHAVIOR_ALLOW_DOUBLE_RANGE_OP_IN_CC
258            = onig_sys::ONIG_SYN_ALLOW_DOUBLE_RANGE_OP_IN_CC;
259        /// `[,-,]`
260        const SYNTAX_BEHAVIOR_WARN_CC_OP_NOT_ESCAPED
